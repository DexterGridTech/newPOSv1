# 07-租户与组织域服务设计

## 1. 服务概述

租户与组织域服务（tenant-service）负责接收并维护从外部系统同步的组织主数据，提供组织层级查询和 Platform 隔离服务。

**本服务是主数据的接收方，不负责：**
- 商户入驻审核与审批
- 合同签署与管理
- 证照资质管理

**核心职责**：
- 接收并存储从上游系统同步的组织主数据（Platform/Project/Tenant/Brand/Store/Contract）
- 维护合同驱动的门店-租户关联关系（`active_contract_id`）
- 提供 Platform 隔离的组织层级查询
- 员工账号与权限管理
- 设备绑定管理

**核心聚合根**：
- Platform（平台/集团）
- Project（购物中心项目）
- Tenant（租户/商户）
- Brand（品牌）
- Store（门店）
- Contract（合同）
- Staff（员工）
- Device（设备）

```java
    private LocalDateTime createdAt;

    @Version
    private Integer version;

     * 从上游同步数据创建或更新租户（幂等）

        tenant.tenantId = IdGenerator.generateTenantId();
        tenant.platformId = data.getPlatformId();
        tenant.externalTenantId = data.getExternalTenantId();
        tenant.tenantCode = data.getTenantCode();
        tenant.tenantName = data.getTenantName();
        tenant.companyName = data.getCompanyName();
        tenant.unifiedSocialCreditCode = data.getUnifiedSocialCreditCode();
        tenant.legalRepresentative = data.getLegalRepresentative();
        tenant.contactPerson = data.getContactPerson();
        tenant.contactPhone = data.getContactPhone();
        tenant.contactEmail = data.getContactEmail();
        tenant.tenantStatus = TenantStatus.ACTIVE;
        tenant.syncedAt = LocalDateTime.now();
        tenant.createdAt = LocalDateTime.now();
        return tenant;
    }

    /**
     * 更新同步数据（已存在时调用）
     */
    public void updateFrom(TenantSyncData data) {
        this.tenantName = data.getTenantName();
        this.companyName = data.getCompanyName();
        this.legalRepresentative = data.getLegalRepresentative();
        this.contactPerson = data.getContactPerson();
        this.contactPhone = data.getContactPhone();
        this.contactEmail = data.getContactEmail();
        this.syncedAt = LocalDateTime.now();
    }

    /**
     * 暂停（由上游状态变更触发）
     */
    public void suspend(String reason) {
        if (this.tenantStatus != TenantStatus.ACTIVE) {
            throw new IllegalStateException("只有激活状态的租户可以暂停");
        }
        this.tenantStatus = TenantStatus.SUSPENDED;
        this.addDomainEvent(new TenantSuspendedEvent(this, reason));
    }
}
```

```java
@Entity
@Table(name = "stores",
    uniqueConstraints = @UniqueConstraint(
        columnNames = {"platform_id", "project_id", "unit_no"}

public class Store {

    @Id
    private String storeId;

    private String platformId;          // 所属 Platform

    private String activeContractId;    // 当前生效合同 ID
    private String tenantId;            // 当前经营租户（快照自 active_contract）
    private String brandId;             // 当前经营品牌（快照自 active_contract）

    @Enumerated(EnumType.STRING)
    private BusinessFormat businessFormat;

    private String addressDetail;       // 购物中心内的门店位置描述（如 B1-018）
    private String storePhone;

    @Enumerated(EnumType.STRING)
    private StoreStatus storeStatus;    // VACANT | PREPARING | OPERATING | SUSPENDED | CLOSED

    private LocalDateTime syncedAt;
    private LocalDateTime createdAt;

    @Version
    private Integer version;

    /**
     * 从上游同步数据创建或更新门店（幂等）
     * 幂等键：platform_id + project_id + unit_no
     */
    public static Store syncFrom(StoreSyncData data) {
        Store store = new Store();
        store.storeId = IdGenerator.generateStoreId();
        store.platformId = data.getPlatformId();
        store.projectId = data.getProjectId();
        store.unitNo = data.getUnitNo();
        store.externalStoreId = data.getExternalStoreId();
        store.storeCode = data.getStoreCode();
        store.storeName = data.getStoreName();
        store.businessFormat = data.getBusinessFormat();
        store.addressDetail = data.getAddressDetail();
        store.storePhone = data.getStorePhone();
        store.storeStatus = StoreStatus.VACANT;
        store.syncedAt = LocalDateTime.now();
        store.createdAt = LocalDateTime.now();
        store.addDomainEvent(new StoreSyncedEvent(store));
        return store;
    }

    /**
     * 激活合同：更新 active_contract 快照
     * 由 ContractActivated 事件触发
     */
    public void activateContract(Contract contract) {
        this.activeContractId = contract.getContractId();
        this.tenantId = contract.getTenantId();
        this.brandId = contract.getBrandId();
        this.storeStatus = StoreStatus.PREPARING;
        this.addDomainEvent(new StoreContractActivatedEvent(this, contract));
    }

    /**
     * 合同到期或终止：清除 active_contract 快照
     */
    public void clearActiveContract(String contractId) {
        if (!contractId.equals(this.activeContractId)) return;
        this.activeContractId = null;
        this.tenantId = null;
        this.brandId = null;
        this.storeStatus = StoreStatus.VACANT;
        this.addDomainEvent(new StoreContractClearedEvent(this, contractId));
    }

    /**
     * 开业
     */
    public void open() {
        if (this.storeStatus != StoreStatus.PREPARING) {
            throw new IllegalStateException("只有筹备中的门店可以开业");
        }
        this.storeStatus = StoreStatus.OPERATING;
        this.addDomainEvent(new StoreOpenedEvent(this));
    }

    /**
     * 暂停营业
     */
    public void suspend(String reason) {
        if (this.storeStatus != StoreStatus.OPERATING) {
            throw new IllegalStateException("只有营业中的门店可以暂停");
        }
        this.storeStatus = StoreStatus.SUSPENDED;
        this.addDomainEvent(new StoreSuspendedEvent(this, reason));
    }
}
```

```java
@Entity

    private LocalDateTime expiredAt;
    private LocalDateTime terminatedAt;
    private String terminationReason;

    private LocalDateTime syncedAt;
    private LocalDateTime createdAt;

    @Version
    private Integer version;

    /**
     * 从上游同步合同数据
     */
    public static Contract syncFrom(ContractSyncData data) {
        Contract c = new Contract();
        c.contractId = IdGenerator.generateContractId();
        c.platformId = data.getPlatformId();
        c.projectId = data.getProjectId();
        c.storeId = data.getStoreId();
        c.unitNo = data.getUnitNo();
        c.tenantId = data.getTenantId();
        c.brandId = data.getBrandId();
        c.contractNo = data.getContractNo();
        c.startDate = data.getStartDate();
        c.endDate = data.getEndDate();
        c.contractStatus = ContractStatus.PENDING;
        c.syncedAt = LocalDateTime.now();
        c.createdAt = LocalDateTime.now();
        return c;
    }

    /**
     * 激活合同（生效日到达时触发）
     */
    public void activate() {
        if (this.contractStatus != ContractStatus.PENDING) {
            throw new IllegalStateException("只有 PENDING 状态的合同可以激活");
        }
        this.contractStatus = ContractStatus.ACTIVE;
        this.activatedAt = LocalDateTime.now();
        this.addDomainEvent(new ContractActivatedEvent(this));
    }

    /**
     * 合同到期
     */
    public void expire() {
        if (this.contractStatus != ContractStatus.ACTIVE) {
            throw new IllegalStateException("只有 ACTIVE 状态的合同可以到期");
        }
        this.contractStatus = ContractStatus.EXPIRED;
        this.expiredAt = LocalDateTime.now();
        this.addDomainEvent(new ContractExpiredEvent(this));
    }

    /**
     * 提前终止合同
     */
    public void terminate(String reason) {
        if (this.contractStatus != ContractStatus.ACTIVE) {
            throw new IllegalStateException("只有 ACTIVE 状态的合同可以终止");
        }
        this.contractStatus = ContractStatus.TERMINATED;
        this.terminatedAt = LocalDateTime.now();
        this.terminationReason = reason;
        this.addDomainEvent(new ContractTerminatedEvent(this));
    }
}
```

```java
     * 登录
     */
    public void login() {

            throw new IllegalStateException("账号未激活或已锁定");

    }

        if (!PasswordUtil.verify(oldPassword, this.passwordHash)) {

        this.passwordExpireAt = LocalDateTime.now().plusMonths(3);

     */
    public void lock() {
```

### 3.1 MasterDataSyncService（主数据同步服务）

```java
@Service
@Transactional
public class MasterDataSyncService {

    @Autowired private PlatformRepository platformRepository;
    @Autowired private ProjectRepository projectRepository;
    @Autowired private TenantRepository tenantRepository;
    @Autowired private BrandRepository brandRepository;
    @Autowired private StoreRepository storeRepository;
    @Autowired private ContractRepository contractRepository;
    @Autowired private DomainEventPublisher eventPublisher;

    /**
     * 同步 Platform（幂等，按 platform_code 去重）
     */
    public SyncResult syncPlatform(PlatformSyncData data) {
        Optional<Platform> existing = platformRepository.findByPlatformCode(data.getPlatformCode());
        Platform platform;
        if (existing.isPresent()) {
            platform = existing.get();
            platform.updateFrom(data);
        } else {
            platform = Platform.syncFrom(data);
        }
        platformRepository.save(platform);
        eventPublisher.publish(platform.getDomainEvents());
        return SyncResult.ok(platform.getPlatformId());
    }

    /**
     * 同步 Tenant（幂等，按 platform_id + unified_social_credit_code 去重）
     */
    public SyncResult syncTenant(TenantSyncData data) {
        Optional<Tenant> existing = tenantRepository
            .findByPlatformIdAndUnifiedSocialCreditCode(
                data.getPlatformId(), data.getUnifiedSocialCreditCode());
        Tenant tenant;
        if (existing.isPresent()) {
            tenant = existing.get();
            tenant.updateFrom(data);
        } else {
            tenant = Tenant.syncFrom(data);
        }
        tenantRepository.save(tenant);
        eventPublisher.publish(tenant.getDomainEvents());
        return SyncResult.ok(tenant.getTenantId());
    }

    /**
     * 同步 Brand（幂等，按 platform_id + external_brand_id 去重）
     */
    public SyncResult syncBrand(BrandSyncData data) {
        Optional<Brand> existing = brandRepository
            .findByPlatformIdAndExternalBrandId(data.getPlatformId(), data.getExternalBrandId());
        Brand brand;
        if (existing.isPresent()) {
            brand = existing.get();
            brand.updateFrom(data);
        } else {
            brand = Brand.syncFrom(data);
        }
        brandRepository.save(brand);
        eventPublisher.publish(brand.getDomainEvents());
        return SyncResult.ok(brand.getBrandId());
    }

    /**
     * 同步 Store（幂等，按 platform_id + project_id + unit_no 去重）
     */
    public SyncResult syncStore(StoreSyncData data) {
        Optional<Store> existing = storeRepository
            .findByPlatformIdAndProjectIdAndUnitNo(
                data.getPlatformId(), data.getProjectId(), data.getUnitNo());
        Store store;
        if (existing.isPresent()) {
            store = existing.get();
            store.updateFrom(data);
        } else {
            store = Store.syncFrom(data);
        }
        storeRepository.save(store);
        eventPublisher.publish(store.getDomainEvents());
        return SyncResult.ok(store.getStoreId());
    }

    /**
     * 同步 Contract（幂等，按 platform_id + contract_no 去重）
     * 合同生效日到达时，触发 activate()；到期时触发 expire()
     */
    public SyncResult syncContract(ContractSyncData data) {
        Optional<Contract> existing = contractRepository
            .findByPlatformIdAndContractNo(data.getPlatformId(), data.getContractNo());
        Contract contract;
        if (existing.isPresent()) {
            contract = existing.get();
            contract.updateFrom(data);
        } else {
            contract = Contract.syncFrom(data);
        }
        contractRepository.save(contract);

        // 如果合同生效日 <= 今天，且状态为 PENDING，则激活
        if (contract.getContractStatus() == ContractStatus.PENDING
                && !contract.getStartDate().isAfter(LocalDate.now())) {
            contract.activate();
            // 级联更新 Store 的 active_contract 快照
            Store store = storeRepository.findById(contract.getStoreId()).orElseThrow();
            store.activateContract(contract);
            storeRepository.save(store);
        }

        contractRepository.save(contract);
        eventPublisher.publish(contract.getDomainEvents());
        return SyncResult.ok(contract.getContractId());
    }
}
```

```java
@Service
@Transactional
public class StoreOperationService {

    @Autowired private StoreRepository storeRepository;
    @Autowired private DomainEventPublisher eventPublisher;

     * 门店开业
     */
    public void openStore(String storeId) {
        Store store = storeRepository.findById(storeId)

        store.open();
        storeRepository.save(store);

     * 暂停营业

    public void suspendStore(String storeId, String reason) {

            .orElseThrow(() -> new StoreNotFoundException(storeId));
        store.suspend(reason);
        storeRepository.save(store);
        eventPublisher.publish(store.getDomainEvents());
    }

    public List<StoreDTO> listStoresByPlatform(String platformId, StoreQueryFilter filter) {
        return storeRepository.findByPlatformIdAndFilter(platformId, filter)
            .stream().map(StoreDTO::from).collect(Collectors.toList());
    }
```

```java
@Service
@Transactional
public class StaffApplicationService {

    @Autowired
    private StaffRepository staffRepository;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 创建员工
     */
    public CreateStaffResponse createStaff(CreateStaffRequest request) {
        // 检查用户名是否已存在
        if (staffRepository.existsByUsername(request.getUsername())) {
            throw new UsernameAlreadyExistsException(request.getUsername());
        }

        // 创建员工
        CreateStaffCommand command = CreateStaffCommand.builder()
            .staffCode(generateStaffCode())
            .staffName(request.getStaffName())
            .tenantId(request.getTenantId())
            .storeId(request.getStoreId())
            .username(request.getUsername())
            .password(request.getPassword())
            .phone(request.getPhone())
            .email(request.getEmail())
            .position(request.getPosition())
            .department(request.getDepartment())
            .roleIds(request.getRoleIds())
            .build();

        Staff staff = Staff.create(command);

        // 保存员工
        staffRepository.save(staff);

        // 发布事件
        eventPublisher.publish(staff.getDomainEvents());

        return CreateStaffResponse.builder()
            .staffId(staff.getStaffId())
            .staffCode(staff.getStaffCode())
            .build();
    }

    /**
     * 员工登录（JWT认证）
     */
    public LoginResponse login(LoginRequest request) {
        Staff staff = staffRepository.findByUsername(request.getUsername())
            .orElseThrow(() -> new InvalidCredentialsException());

        if (!PasswordUtil.verify(request.getPassword(), staff.getPasswordHash())) {
            throw new InvalidCredentialsException();
        }

        staff.login();
        staffRepository.save(staff);

        // 生成JWT Token
        String token = JwtUtil.generateToken(staff);

        return LoginResponse.builder()
            .token(token)
            .staffId(staff.getStaffId())
            .staffName(staff.getStaffName())
            .build();
    }

    private String generateStaffCode() {
        return "E" + System.currentTimeMillis();
    }
}
```

### 4.2 DeviceManagementService（基于 TDP 的设备命令推送）

```java
@Service
public class DeviceManagementService {

    @Autowired
    private DeviceCommandRepository commandRepository;

    @Autowired
    private TDPWriteClient tdpWriteClient;

    @Autowired
    private ApplicationEventPublisher eventPublisher;

    /**
     * 发送设备命令（幂等）
     * 通过 TDP 推送到设备终端
     */
    public DeviceCommand sendCommand(String deviceId, CommandType type,
                                      String payload, String idempotencyKey) {
        // 幂等检查
        Optional<DeviceCommand> existing = commandRepository
            .findByIdempotencyKey(idempotencyKey);
        if (existing.isPresent()) {
            return existing.get();
        }

        DeviceCommand command = new DeviceCommand();
        command.setCommandId(UUID.randomUUID().toString());
        command.setDeviceId(deviceId);
        command.setCommandType(type);
        command.setPayload(payload);
        command.setIdempotencyKey(idempotencyKey);
        command.setStatus(CommandStatus.PENDING);
        command.setMaxRetry(3);
        command.setExpireAt(LocalDateTime.now().plusMinutes(5));
        commandRepository.save(command);

        // 通过 TDP 推送到设备
        ProjectionWriteRequest request = ProjectionWriteRequest.builder()
            .tenantId(command.getTenantId())
            .topic("remote.control")  // 远程控制命令 Topic
            .itemKey("device-cmd:" + command.getCommandId())
            .scopeType("terminal")
            .scopeId("device:" + deviceId)
            .payload(buildCommandPayload(command))
            .sourceSystem("tenant-service")
            .build();

        tdpWriteClient.writeProjection(request);

        command.setStatus(CommandStatus.SENT);
        command.setSentAt(LocalDateTime.now());
        commandRepository.save(command);

        return command;
    }

    /**
     * 处理设备回执
     */
    public void processAcknowledgement(String commandId, boolean success, String errorMsg) {
        DeviceCommand command = commandRepository.findById(commandId)
            .orElseThrow(() -> new NotFoundException("命令不存在: " + commandId));

        if (success) {
            command.setStatus(CommandStatus.ACKNOWLEDGED);
            command.setAcknowledgedAt(LocalDateTime.now());
        } else {
            command.setStatus(CommandStatus.FAILED);
            command.setErrorMessage(errorMsg);
        }
        commandRepository.save(command);

        eventPublisher.publishEvent(new DeviceCommandAcknowledgedEvent(command));
    }

    /**
     * 超时重试（定时任务调用）
     */
    @Scheduled(fixedDelay = 10000)
    public void retryTimeoutCommands() {
        List<DeviceCommand> timeouts = commandRepository
            .findSentCommandsOlderThan(LocalDateTime.now().minusSeconds(30));

        for (DeviceCommand cmd : timeouts) {
            if (cmd.getRetryCount() < cmd.getMaxRetry()) {
                cmd.setRetryCount(cmd.getRetryCount() + 1);

                // 通过 TDP 重新推送
                ProjectionWriteRequest request = ProjectionWriteRequest.builder()
                    .tenantId(cmd.getTenantId())
                    .topic("remote.control")
                    .itemKey("device-cmd:" + cmd.getCommandId())
                    .scopeType("terminal")
                    .scopeId("device:" + cmd.getDeviceId())
                    .payload(buildCommandPayload(cmd))
                    .sourceSystem("tenant-service")
                    .build();

                tdpWriteClient.writeProjection(request);
                commandRepository.save(cmd);
            } else {
                cmd.setStatus(CommandStatus.TIMEOUT);
                commandRepository.save(cmd);
                eventPublisher.publishEvent(new DeviceCommandTimeoutEvent(cmd));
            }
        }
    }

    private Map<String, Object> buildCommandPayload(DeviceCommand command) {
        Map<String, Object> payload = new HashMap<>();
        payload.put("commandId", command.getCommandId());
        payload.put("commandType", command.getCommandType().name());
        payload.put("payload", command.getPayload());
        payload.put("expireAt", command.getExpireAt().toString());
        return payload;
    }
```

### 4.3 离线操作日志处理（离线同步）

```java
@Service
public class OfflineOperationService {

    @Autowired
    private OfflineOperationLogRepository logRepository;

    /**
     * 设备上线后批量同步离线操作
     */
    public SyncResult syncOfflineOperations(String deviceId,
                                             List<OfflineOperationDTO> operations) {
        int success = 0, conflict = 0;

        for (OfflineOperationDTO op : operations) {
            OfflineOperationLog log = new OfflineOperationLog();
            log.setLogId(UUID.randomUUID().toString());
            log.setDeviceId(deviceId);
            log.setOperationType(op.getOperationType());
            log.setOperationPayload(op.getPayload());
            log.setOperatedAt(op.getOperatedAt());
            log.setIdempotencyKey(op.getIdempotencyKey());

            try {
                applyOperation(op);
                log.setSyncStatus(SyncStatus.SYNCED);
                success++;
            } catch (ConflictException e) {
                log.setSyncStatus(SyncStatus.CONFLICT);
                log.setConflictReason(e.getMessage());
                conflict++;
            }
            logRepository.save(log);
        }

        return new SyncResult(success, conflict);
    }

    private void applyOperation(OfflineOperationDTO op) {
        // 根据操作类型路由到对应的应用服务
        switch (op.getOperationType()) {
            case "CREATE_ORDER" -> orderAppService.createOrderFromOffline(op.getPayload());
            case "UPDATE_STOCK" -> stockAppService.updateStockFromOffline(op.getPayload());
            default -> throw new UnsupportedOperationException("不支持的离线操作类型: " + op.getOperationType());
        }
    }
}
```

## 7. 本文结论

租户与组织域服务设计展示了以**主数据接收方**为定位的完整组织管理实现：

- **领域模型**：Platform、Contract 是新增的核心聚合根；Tenant、Store 去除了入驻/审核字段，以 syncedAt 和 externalId 取代
- **主数据同步**：MasterDataSyncService 提供幂等同步入口，按业务唯一键（platform_id + 业务主键）去重
- **合同驱动**：Store 的 tenantId/brandId 由 active_contract 快照驱动，合同激活/到期/终止时自动同步门店状态
- **Platform 隔离**：所有查询强制携带 platform_id，数据库层通过联合唯一约束物理隔离
- **员工管理**：Staff 聚合根设计、JWT认证、角色权限管理保持不变
- **设备命令**：DeviceCommand 实体、基于 TDP 推送、发送/回执/超时重试、离线操作同步
- **接口设计**：主数据同步 REST API（`/api/v1/sync/*`）+ 门店运营 REST API + 员工登录接口
- **数据库设计**：新增 platforms、projects、contracts 表；tenants 表移除审核字段；stores 表以 platform_id + project_id + unit_no 为唯一键
