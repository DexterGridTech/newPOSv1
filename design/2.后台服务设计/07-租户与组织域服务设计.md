# 07-租户与组织域服务设计

## 1. 服务概述

租户与组织域服务（tenant-service）负责租户管理、组织架构管理、员工管理、设备管理等功能。

**核心职责**：
- 租户入驻与审核
- 品牌与门店管理
- 经营主体管理
- 员工账号与权限管理
- 设备绑定管理

**核心聚合根**：
- Tenant（租户）
- Brand（品牌）
- Store（门店）
- Staff（员工）
- Device（设备）

```java
@Entity
@Table(name = "tenants")
public class Tenant {

    @Id
    private String tenantId;

    private String tenantCode;
    private String tenantName;

    @Enumerated(EnumType.STRING)
    private TenantType tenantType;

    private String companyName;
    private String unifiedSocialCreditCode;
    private String legalRepresentative;

    private String contactPerson;
    private String contactPhone;
    private String contactEmail;

    private String businessLicenseNo;
    private String businessLicenseUrl;
    private String foodLicenseNo;
    private String foodLicenseUrl;
    private LocalDate foodLicenseExpireDate;

    @Enumerated(EnumType.STRING)
    private OnboardingStatus onboardingStatus;

    @Enumerated(EnumType.STRING)
    private ReviewStatus reviewStatus;

    private LocalDateTime submittedAt;
    private LocalDateTime reviewedAt;
    private String reviewedBy;
    private String reviewComment;

    @Enumerated(EnumType.STRING)
    private AccountStatus accountStatus;

    private LocalDateTime createdAt;

    @Version
    private Integer version;

    /**
     * 提交审核
     */
    public void submitForReview() {
        if (this.onboardingStatus != OnboardingStatus.DRAFT) {
            throw new IllegalStateException("只有草稿状态可以提交审核");
        }

        validateTenantInfo();

        this.onboardingStatus = OnboardingStatus.PENDING_REVIEW;
        this.reviewStatus = ReviewStatus.PENDING;
        this.submittedAt = LocalDateTime.now();

        this.addDomainEvent(new TenantSubmittedForReviewEvent(this));
    }

    /**
     * 审核通过
     */
    public void approve(String reviewerId, String comment) {
        if (this.reviewStatus != ReviewStatus.PENDING) {
            throw new IllegalStateException("只有待审核状态可以审核");
        }

        this.reviewStatus = ReviewStatus.APPROVED;
        this.onboardingStatus = OnboardingStatus.APPROVED;
        this.reviewedAt = LocalDateTime.now();
        this.reviewedBy = reviewerId;
        this.reviewComment = comment;

        this.addDomainEvent(new TenantApprovedEvent(this));
    }

    /**
     * 激活租户
     */
    public void activate() {
        if (this.onboardingStatus != OnboardingStatus.APPROVED) {
            throw new IllegalStateException("只有审核通过的租户可以激活");
        }

        this.accountStatus = AccountStatus.ACTIVE;
        this.addDomainEvent(new TenantActivatedEvent(this));
    }

    /**
     * 暂停租户
     */
    public void suspend(String reason) {
        if (this.accountStatus != AccountStatus.ACTIVE) {
            throw new IllegalStateException("只有激活状态的租户可以暂停");
        }

        this.accountStatus = AccountStatus.SUSPENDED;
        this.addDomainEvent(new TenantSuspendedEvent(this, reason));
    }

    private void validateTenantInfo() {
        if (StringUtils.isEmpty(companyName)) {
            throw new IllegalStateException("企业名称不能为空");
        }
        if (StringUtils.isEmpty(unifiedSocialCreditCode)) {
            throw new IllegalStateException("统一社会信用代码不能为空");
        }
        if (StringUtils.isEmpty(businessLicenseNo)) {
            throw new IllegalStateException("营业执照号不能为空");
        }
    }
}
```

```java
    }
```

```java
     * 修改密码
     */
    public void changePassword(String oldPassword, String newPassword) {
        if (!PasswordUtil.verify(oldPassword, this.passwordHash)) {
            throw new IllegalArgumentException("原密码不正确");
        }

        this.passwordHash = PasswordUtil.hash(newPassword);
        this.passwordExpireAt = LocalDateTime.now().plusMonths(3);
    }

    /**
     * 锁定账号
     */
    public void lock() {
        this.accountStatus = AccountStatus.LOCKED;
        this.addDomainEvent(new StaffLockedEvent(this));
    }
```

### 3.1 TenantApplicationService（租户入驻审核流程）

```java
@Service
@Transactional
public class TenantApplicationService {

    @Autowired
    private TenantRepository tenantRepository;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 提交租户申请
     */
    public SubmitTenantApplicationResponse submitTenantApplication(
        SubmitTenantApplicationRequest request
    ) {
        // 验证请求
        validateTenantApplication(request);

        // 创建租户
        Tenant tenant = Tenant.builder()
            .tenantId(IdGenerator.generateTenantId())
            .tenantCode(generateTenantCode())
            .tenantName(request.getTenantName())
            .tenantType(request.getTenantType())
            .companyName(request.getCompanyName())
            .unifiedSocialCreditCode(request.getUnifiedSocialCreditCode())
            .legalRepresentative(request.getLegalRepresentative())
            .contactPerson(request.getContactPerson())
            .contactPhone(request.getContactPhone())
            .contactEmail(request.getContactEmail())
            .businessLicenseNo(request.getBusinessLicenseNo())
            .businessLicenseUrl(request.getBusinessLicenseUrl())
            .foodLicenseNo(request.getFoodLicenseNo())
            .foodLicenseUrl(request.getFoodLicenseUrl())
            .onboardingStatus(OnboardingStatus.DRAFT)
            .accountStatus(AccountStatus.INACTIVE)
            .createdAt(LocalDateTime.now())
            .build();

        // 保存租户
        tenantRepository.save(tenant);

        // 提交审核
        tenant.submitForReview();
        tenantRepository.save(tenant);

        // 发布事件
        eventPublisher.publish(tenant.getDomainEvents());

        return SubmitTenantApplicationResponse.builder()
            .tenantId(tenant.getTenantId())
            .build();
    }

    /**
     * 审核租户
     */
    public void reviewTenant(String tenantId, boolean approved, String comment) {
        Tenant tenant = tenantRepository.findById(tenantId)
            .orElseThrow(() -> new TenantNotFoundException(tenantId));

        if (approved) {
            tenant.approve("SYSTEM", comment);
        } else {
            tenant.reject(comment);
        }

        tenantRepository.save(tenant);
        eventPublisher.publish(tenant.getDomainEvents());
    }

    /**
     * 激活租户
     */
    public void activateTenant(String tenantId) {
        Tenant tenant = tenantRepository.findById(tenantId)
            .orElseThrow(() -> new TenantNotFoundException(tenantId));

        tenant.activate();

        tenantRepository.save(tenant);
        eventPublisher.publish(tenant.getDomainEvents());
    }

    private String generateTenantCode() {
        return "T" + System.currentTimeMillis();
    }
}
```

```java
            .password(request.getPassword())
            .phone(request.getPhone())
            .email(request.getEmail())
            .position(request.getPosition())
            .department(request.getDepartment())
            .roleIds(request.getRoleIds())
            .build();

        Staff staff = Staff.create(command);

        // 保存员工
        staffRepository.save(staff);

        // 发布事件
        eventPublisher.publish(staff.getDomainEvents());

        return CreateStaffResponse.builder()
            .staffId(staff.getStaffId())
            .staffCode(staff.getStaffCode())
            .build();
    }

    /**
     * 员工登录（JWT认证）
     */
    public LoginResponse login(LoginRequest request) {
        Staff staff = staffRepository.findByUsername(request.getUsername())
            .orElseThrow(() -> new InvalidCredentialsException());

        if (!PasswordUtil.verify(request.getPassword(), staff.getPasswordHash())) {
            throw new InvalidCredentialsException();
        }

        staff.login();
        staffRepository.save(staff);

        // 生成JWT Token
        String token = JwtUtil.generateToken(staff);

        return LoginResponse.builder()
            .token(token)
            .staffId(staff.getStaffId())
            .staffName(staff.getStaffName())
            .build();
    }

    private String generateStaffCode() {
        return "E" + System.currentTimeMillis();
    }
}
```

### 4.2 DeviceManagementService（基于 TDP 的设备命令推送）

```java
@Service
public class DeviceManagementService {

    @Autowired
    private DeviceCommandRepository commandRepository;

    @Autowired
    private TDPWriteClient tdpWriteClient;

    @Autowired
    private ApplicationEventPublisher eventPublisher;

    /**
     * 发送设备命令（幂等）
     * 通过 TDP 推送到设备终端
     */
    public DeviceCommand sendCommand(String deviceId, CommandType type,
                                      String payload, String idempotencyKey) {
        // 幂等检查
        Optional<DeviceCommand> existing = commandRepository
            .findByIdempotencyKey(idempotencyKey);
        if (existing.isPresent()) {
            return existing.get();
        }

        DeviceCommand command = new DeviceCommand();
        command.setCommandId(UUID.randomUUID().toString());
        command.setDeviceId(deviceId);
        command.setCommandType(type);
        command.setPayload(payload);
        command.setIdempotencyKey(idempotencyKey);
        command.setStatus(CommandStatus.PENDING);
        command.setMaxRetry(3);
        command.setExpireAt(LocalDateTime.now().plusMinutes(5));
        commandRepository.save(command);

        // 通过 TDP 推送到设备
        ProjectionWriteRequest request = ProjectionWriteRequest.builder()
            .tenantId(command.getTenantId())
            .topic("remote.control")  // 远程控制命令 Topic
            .itemKey("device-cmd:" + command.getCommandId())
            .scopeType("terminal")
            .scopeId("device:" + deviceId)
            .payload(buildCommandPayload(command))
            .sourceSystem("tenant-service")
            .build();

        tdpWriteClient.writeProjection(request);

        command.setStatus(CommandStatus.SENT);
        command.setSentAt(LocalDateTime.now());
        commandRepository.save(command);

        return command;
    }

    /**
     * 处理设备回执
     */
    public void processAcknowledgement(String commandId, boolean success, String errorMsg) {
        DeviceCommand command = commandRepository.findById(commandId)
            .orElseThrow(() -> new NotFoundException("命令不存在: " + commandId));

        if (success) {
            command.setStatus(CommandStatus.ACKNOWLEDGED);
            command.setAcknowledgedAt(LocalDateTime.now());
        } else {
            command.setStatus(CommandStatus.FAILED);
            command.setErrorMessage(errorMsg);
        }
        commandRepository.save(command);

        eventPublisher.publishEvent(new DeviceCommandAcknowledgedEvent(command));
    }

    /**
     * 超时重试（定时任务调用）
     */
    @Scheduled(fixedDelay = 10000)
    public void retryTimeoutCommands() {
        List<DeviceCommand> timeouts = commandRepository
            .findSentCommandsOlderThan(LocalDateTime.now().minusSeconds(30));

        for (DeviceCommand cmd : timeouts) {
            if (cmd.getRetryCount() < cmd.getMaxRetry()) {
                cmd.setRetryCount(cmd.getRetryCount() + 1);

                // 通过 TDP 重新推送
                ProjectionWriteRequest request = ProjectionWriteRequest.builder()
                    .tenantId(cmd.getTenantId())
                    .topic("remote.control")
                    .itemKey("device-cmd:" + cmd.getCommandId())
                    .scopeType("terminal")
                    .scopeId("device:" + cmd.getDeviceId())
                    .payload(buildCommandPayload(cmd))
                    .sourceSystem("tenant-service")
                    .build();

                tdpWriteClient.writeProjection(request);
                commandRepository.save(cmd);
            } else {
                cmd.setStatus(CommandStatus.TIMEOUT);
                commandRepository.save(cmd);
                eventPublisher.publishEvent(new DeviceCommandTimeoutEvent(cmd));
            }
        }
    }

    private Map<String, Object> buildCommandPayload(DeviceCommand command) {
        Map<String, Object> payload = new HashMap<>();
        payload.put("commandId", command.getCommandId());
        payload.put("commandType", command.getCommandType().name());
        payload.put("payload", command.getPayload());
        payload.put("expireAt", command.getExpireAt().toString());
        return payload;
    }
}
```

### 4.3 离线操作日志处理（离线同步）

```java
@Service
public class OfflineOperationService {

    @Autowired
    private OfflineOperationLogRepository logRepository;

    /**
     * 设备上线后批量同步离线操作
     */
    public SyncResult syncOfflineOperations(String deviceId,
                                             List<OfflineOperationDTO> operations) {
        int success = 0, conflict = 0;

        for (OfflineOperationDTO op : operations) {
            OfflineOperationLog log = new OfflineOperationLog();
            log.setLogId(UUID.randomUUID().toString());
            log.setDeviceId(deviceId);
            log.setOperationType(op.getOperationType());
            log.setOperationPayload(op.getPayload());
            log.setOperatedAt(op.getOperatedAt());
            log.setIdempotencyKey(op.getIdempotencyKey());

            try {
                applyOperation(op);
                log.setSyncStatus(SyncStatus.SYNCED);
                success++;
            } catch (ConflictException e) {
                log.setSyncStatus(SyncStatus.CONFLICT);
                log.setConflictReason(e.getMessage());
                conflict++;
            }
            logRepository.save(log);
        }

        return new SyncResult(success, conflict);
    }

    private void applyOperation(OfflineOperationDTO op) {
        // 根据操作类型路由到对应的应用服务
        switch (op.getOperationType()) {
            case "CREATE_ORDER" -> orderAppService.createOrderFromOffline(op.getPayload());
            case "UPDATE_STOCK" -> stockAppService.updateStockFromOffline(op.getPayload());
            default -> throw new UnsupportedOperationException("不支持的离线操作类型: " + op.getOperationType());
        }
    }
}
```

```java
        @RequestBody List<OfflineOperationDTO> operations

        SyncResult result = offlineOperationService.syncOfflineOperations(deviceId, operations);
        return Result.success(result);
    }
}
```
