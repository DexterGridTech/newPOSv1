# 06-结算与对账域服务设计

## 1. 服务概述

结算与对账域服务（settlement-service）负责结算事实生成、清分计算、账单生成、对账任务执行等功能。

**核心职责**：
- 结算事实生成
- 多方清分计算
- 结算账单生成
- 对账任务执行
- 差异处理

**核心聚合根**：
- SettlementFact（结算事实）
- SettlementCycle（结算周期）
- SettlementBill（结算账单）
- ReconciliationTask（对账任务）

```java
public class SettlementFact {

    private String settlementCycleId;
    private LocalDateTime settledAt;

    @Enumerated(EnumType.STRING)
    private ReconciliationStatus reconciliationStatus;

    private LocalDateTime reconciledAt;
    private Boolean hasDiff;
    private Long diffAmount;

    private LocalDateTime createdAt;

    @Version
    private Integer version;

    /**
     * 创建结算事实
     */
    public static SettlementFact create(
        SaleOrder order,
        CommissionRule rule,
        CommissionResult commissionResult
    ) {
        SettlementFact fact = new SettlementFact();
        fact.settlementFactId = IdGenerator.generateSettlementFactId();
        fact.orderId = order.getOrderId();
        fact.storeId = order.getStoreId();
        fact.tenantId = order.getTenantId();
        fact.channelType = order.getChannelType();

        // 金额信息
        fact.totalAmount = order.getTotalAmount();
        fact.discountAmount = order.getDiscountAmount();
        fact.payableAmount = order.getPayableAmount();
        fact.paidAmount = order.getPaidAmount();
        fact.refundAmount = order.getRefundAmount();
        fact.actualSettlementAmount = order.getPaidAmount() - order.getRefundAmount();

        // 清分规则快照
        fact.commissionRuleId = rule.getCommissionRuleId();
        fact.commissionRuleSnapshot = CommissionRuleSnapshot.from(rule);

        // 清分结果
        fact.grossRevenue = fact.actualSettlementAmount;
        fact.platformCommission = commissionResult.getPlatformCommission();
        fact.channelCommission = commissionResult.getChannelCommission();
        fact.deliveryFee = commissionResult.getDeliveryFee();
        fact.paymentFee = commissionResult.getPaymentFee();
        fact.merchantReceivable = commissionResult.getMerchantReceivable();

        fact.settlementStatus = SettlementStatus.PENDING;
        fact.reconciliationStatus = ReconciliationStatus.NOT_RECONCILED;
        fact.createdAt = LocalDateTime.now();

        fact.addDomainEvent(new SettlementFactCreatedEvent(fact));

        return fact;
    }

    /**
     * 关联到结算周期
     */
    public void associateWithCycle(String cycleId) {
        this.settlementCycleId = cycleId;
    }

    /**
     * 标记已结算
     */
    public void markSettled() {
        if (this.settlementStatus != SettlementStatus.BILL_GENERATED) {
            throw new IllegalStateException("结算状态不正确");
        }

        this.settlementStatus = SettlementStatus.SETTLED;
        this.settledAt = LocalDateTime.now();
    }

    /**
     * 标记已对账
     */
    public void markReconciled(boolean hasDiff, Long diffAmount) {
        this.reconciliationStatus = hasDiff ?
            ReconciliationStatus.DIFF_FOUND : ReconciliationStatus.RECONCILED;
        this.reconciledAt = LocalDateTime.now();
        this.hasDiff = hasDiff;
        this.diffAmount = diffAmount;
    }
}
```

```java
@Entity
@Table(name = "settlement_cycles")
public class SettlementCycle {

    @Id
    private String settlementCycleId;

    @Enumerated(EnumType.STRING)
    private CycleType cycleType;

    private String cycleName;
    private LocalDate startDate;
    private LocalDate endDate;
    private LocalDateTime cutOffTime;

    // 统计信息
    private Integer totalOrders;
    private Long totalRevenue;
    private Long totalCommission;
    private Long totalMerchantReceivable;
    private Long totalRefund;

    @Enumerated(EnumType.STRING)
    private CycleStatus cycleStatus;

    private LocalDateTime closedAt;
    private LocalDateTime settledAt;
    private LocalDateTime reconciledAt;

    private String closedBy;
    private String settledBy;

    /**
     * 创建结算周期
     */
    public static SettlementCycle create(CycleType cycleType, LocalDate startDate) {
        SettlementCycle cycle = new SettlementCycle();
        cycle.settlementCycleId = IdGenerator.generateSettlementCycleId();
        cycle.cycleType = cycleType;
        cycle.startDate = startDate;
        cycle.endDate = calculateEndDate(cycleType, startDate);
        cycle.cycleName = generateCycleName(cycleType, startDate);
        cycle.cycleStatus = CycleStatus.OPEN;
        cycle.totalOrders = 0;
        cycle.totalRevenue = 0L;
        cycle.totalCommission = 0L;
        cycle.totalMerchantReceivable = 0L;
        cycle.totalRefund = 0L;

        return cycle;
    }

    /**
     * 关闭周期
     */
    public void close(String operatorId, CycleSummary summary) {
        if (this.cycleStatus != CycleStatus.OPEN) {
            throw new IllegalStateException("周期状态不正确");
        }

        this.cycleStatus = CycleStatus.CLOSED;
        this.closedAt = LocalDateTime.now();
        this.closedBy = operatorId;

        // 更新统计信息
        this.totalOrders = summary.getTotalOrders();
        this.totalRevenue = summary.getTotalRevenue();
        this.totalCommission = summary.getTotalCommission();
        this.totalMerchantReceivable = summary.getTotalMerchantReceivable();
        this.totalRefund = summary.getTotalRefund();

        this.addDomainEvent(new SettlementCycleClosedEvent(this));
    }

    /**
     * 标记账单已生成
     */
    public void markBillGenerated() {
        if (this.cycleStatus != CycleStatus.CLOSED) {
            throw new IllegalStateException("周期状态不正确");
        }

        this.cycleStatus = CycleStatus.BILL_GENERATED;
    }

    /**
     * 标记已结算
     */
    public void markSettled(String operatorId) {
        if (this.cycleStatus != CycleStatus.BILL_GENERATED) {
            throw new IllegalStateException("周期状态不正确");
        }

        this.cycleStatus = CycleStatus.SETTLED;
        this.settledAt = LocalDateTime.now();
        this.settledBy = operatorId;
    }

    private static LocalDate calculateEndDate(CycleType cycleType, LocalDate startDate) {
        switch (cycleType) {
            case DAILY:
                return startDate;
            case WEEKLY:
                return startDate.plusDays(6);
            case MONTHLY:
                return startDate.plusMonths(1).minusDays(1);
            default:
                return startDate;
        }
    }

    private static String generateCycleName(CycleType cycleType, LocalDate startDate) {
        switch (cycleType) {
            case DAILY:
                return startDate.format(DateTimeFormatter.ofPattern("yyyy-MM-dd"));
            case WEEKLY:
                return startDate.format(DateTimeFormatter.ofPattern("yyyy年第W周"));
            case MONTHLY:
                return startDate.format(DateTimeFormatter.ofPattern("yyyy年MM月"));
            default:
                return startDate.toString();
        }
    }
}
```

```java
@Entity
@Table(name = "settlement_bills")
public class SettlementBill {

    @Id
    private String settlementBillId;

    private String settlementCycleId;

    @Enumerated(EnumType.STRING)
    private BillType billType;

    private String billNo;

    // 账单主体
    @Enumerated(EnumType.STRING)
    private SubjectType subjectType;

    private String subjectId;
    private String subjectName;

    // 账单期间
    private LocalDate billStartDate;
    private LocalDate billEndDate;

    // 汇总金额
    private Integer totalOrders;
    private Long grossRevenue;
    private Long totalCommission;
    private Long totalDeliveryFee;
    private Long totalPaymentFee;
    private Long totalRefund;
    private Long netReceivable;

    @OneToMany(mappedBy = "settlementBill", cascade = CascadeType.ALL)
    private List<SettlementBillItem> items;

    @Enumerated(EnumType.STRING)
    private BillStatus billStatus;

    private LocalDateTime confirmedAt;
    private LocalDateTime paidAt;

    private String billFileUrl;
    private String detailFileUrl;

    private LocalDateTime createdAt;
    private String createdBy;

    /**
     * 创建结算账单
     */
    public static SettlementBill create(
        SettlementCycle cycle,
        SubjectType subjectType,
        String subjectId,
        String subjectName,
        List<SettlementFact> facts
    ) {
        SettlementBill bill = new SettlementBill();
        bill.settlementBillId = IdGenerator.generateSettlementBillId();
        bill.settlementCycleId = cycle.getSettlementCycleId();
        bill.billType = determineBillType(subjectType);
        bill.billNo = generateBillNo(cycle, subjectType, subjectId);
        bill.subjectType = subjectType;
        bill.subjectId = subjectId;
        bill.subjectName = subjectName;
        bill.billStartDate = cycle.getStartDate();
        bill.billEndDate = cycle.getEndDate();
        bill.billStatus = BillStatus.DRAFT;
        bill.createdAt = LocalDateTime.now();

        // 计算汇总金额
        bill.totalOrders = facts.size();
        bill.grossRevenue = facts.stream().mapToLong(SettlementFact::getGrossRevenue).sum();
        bill.totalCommission = facts.stream().mapToLong(SettlementFact::getPlatformCommission).sum();
        bill.totalDeliveryFee = facts.stream().mapToLong(SettlementFact::getDeliveryFee).sum();
        bill.totalPaymentFee = facts.stream().mapToLong(SettlementFact::getPaymentFee).sum();
        bill.totalRefund = facts.stream().mapToLong(SettlementFact::getRefundAmount).sum();
        bill.netReceivable = facts.stream().mapToLong(SettlementFact::getMerchantReceivable).sum();

        // 创建账单明细
        bill.items = facts.stream()
            .map(fact -> SettlementBillItem.create(bill, fact))
            .collect(Collectors.toList());

        bill.addDomainEvent(new SettlementBillGeneratedEvent(bill));

        return bill;
    }

    /**
     * 确认账单
     */
    public void confirm() {
        if (this.billStatus != BillStatus.DRAFT) {
            throw new IllegalStateException("账单状态不正确");
        }

        this.billStatus = BillStatus.CONFIRMED;
        this.confirmedAt = LocalDateTime.now();
    }

    /**
     * 标记已支付
     */
    public void markPaid() {
        if (this.billStatus != BillStatus.CONFIRMED) {
            throw new IllegalStateException("账单状态不正确");
        }

        this.billStatus = BillStatus.PAID;
        this.paidAt = LocalDateTime.now();
    }

    private static BillType determineBillType(SubjectType subjectType) {
        switch (subjectType) {
            case STORE:
            case TENANT:
            case ENTITY:
                return BillType.MERCHANT;
            case CHANNEL:
                return BillType.CHANNEL;
            case DELIVERY_PROVIDER:
                return BillType.DELIVERY;
            default:
                return BillType.PLATFORM;
        }
    }

    private static String generateBillNo(
        SettlementCycle cycle,
        SubjectType subjectType,
        String subjectId
    ) {
        return String.format("BILL-%s-%s-%s-%s",
            cycle.getCycleType().name(),
            cycle.getStartDate().format(DateTimeFormatter.ofPattern("yyyyMMdd")),
            subjectType.name(),
            subjectId.substring(0, 8)
        );
    }
}
```

## 3. 应用服务设计

### 3.1 SettlementApplicationService

```java
@Service
@Transactional
public class SettlementApplicationService {

    @Autowired
    private SettlementFactRepository settlementFactRepository;

    @Autowired
    private CommissionCalculationService commissionCalculationService;

    @Autowired
    private SettlementCycleRepository settlementCycleRepository;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 生成结算事实
     */
    public void generateSettlementFact(String orderId) {
        // 1. 查询订单
        SaleOrder order = orderRepository.findById(orderId)
            .orElseThrow(() -> new OrderNotFoundException(orderId));

        // 2. 查询适用的佣金规则
        CommissionRule rule = commissionRuleRepository
            .findApplicableRule(order.getStoreId(), order.getChannelType())
            .orElseThrow(() -> new CommissionRuleNotFoundException());

        // 3. 计算清分结果
        CommissionResult commissionResult = commissionCalculationService
            .calculate(order, rule);

        // 4. 创建结算事实
        SettlementFact fact = SettlementFact.create(order, rule, commissionResult);

        // 5. 关联到当前开放的结算周期
        SettlementCycle currentCycle = settlementCycleRepository
            .findCurrentOpenCycle()
            .orElseGet(() -> createNewCycle());

        fact.associateWithCycle(currentCycle.getSettlementCycleId());

        // 6. 保存结算事实
        settlementFactRepository.save(fact);

        // 7. 发布事件
        eventPublisher.publish(fact.getDomainEvents());
    }

    /**
     * 关闭结算周期
     */
    public void closeSettlementCycle(String cycleId, String operatorId) {
        // 1. 查询结算周期
        SettlementCycle cycle = settlementCycleRepository.findById(cycleId)
            .orElseThrow(() -> new SettlementCycleNotFoundException(cycleId));

        // 2. 统计周期内的结算事实
        CycleSummary summary = calculateCycleSummary(cycleId);

        // 3. 关闭周期
        cycle.close(operatorId, summary);

        // 4. 保存周期
        settlementCycleRepository.save(cycle);

        // 5. 创建新周期
        SettlementCycle newCycle = SettlementCycle.create(
            cycle.getCycleType(),
            cycle.getEndDate().plusDays(1)
        );
        settlementCycleRepository.save(newCycle);

        // 6. 发布事件
        eventPublisher.publish(cycle.getDomainEvents());
    }

    /**
     * 生成结算账单
     */
    public void generateSettlementBills(String cycleId) {
        // 1. 查询结算周期
        SettlementCycle cycle = settlementCycleRepository.findById(cycleId)
            .orElseThrow(() -> new SettlementCycleNotFoundException(cycleId));

        // 2. 查询周期内的所有结算事实
        List<SettlementFact> facts = settlementFactRepository
            .findBySettlementCycleId(cycleId);

        // 3. 按主体分组
        Map<String, List<SettlementFact>> groupedByStore = facts.stream()
            .collect(Collectors.groupingBy(SettlementFact::getStoreId));

        // 4. 为每个门店生成账单
        List<SettlementBill> bills = new ArrayList<>();
        for (Map.Entry<String, List<SettlementFact>> entry : groupedByStore.entrySet()) {
            String storeId = entry.getKey();
            List<SettlementFact> storeFacts = entry.getValue();

            Store store = storeRepository.findById(storeId)
                .orElseThrow(() -> new StoreNotFoundException(storeId));

            SettlementBill bill = SettlementBill.create(
                cycle,
                SubjectType.STORE,
                storeId,
                store.getStoreName(),
                storeFacts
            );

            bills.add(bill);
        }

        // 5. 保存账单
        settlementBillRepository.saveAll(bills);

        // 6. 更新周期状态
        cycle.markBillGenerated();
        settlementCycleRepository.save(cycle);

        // 7. 发布事件
        bills.forEach(bill -> eventPublisher.publish(bill.getDomainEvents()));
    }

    private CycleSummary calculateCycleSummary(String cycleId) {
        List<SettlementFact> facts = settlementFactRepository
            .findBySettlementCycleId(cycleId);

        return CycleSummary.builder()
            .totalOrders(facts.size())
            .totalRevenue(facts.stream().mapToLong(SettlementFact::getGrossRevenue).sum())
            .totalCommission(facts.stream().mapToLong(SettlementFact::getPlatformCommission).sum())
            .totalMerchantReceivable(facts.stream().mapToLong(SettlementFact::getMerchantReceivable).sum())
            .totalRefund(facts.stream().mapToLong(SettlementFact::getRefundAmount).sum())
            .build();
    }

    private SettlementCycle createNewCycle() {
        SettlementCycle cycle = SettlementCycle.create(
            CycleType.DAILY,
            LocalDate.now()
        );
        return settlementCycleRepository.save(cycle);
    }
}
```

### 3.2 CommissionCalculationService（清分计算引擎）

```java
@Service
public class CommissionCalculationService {

    /**
     * 计算清分结果
     */
    public CommissionResult calculate(SaleOrder order, CommissionRule rule) {
        Long grossRevenue = order.getPaidAmount() - order.getRefundAmount();

        // 1. 计算平台佣金
        Long platformCommission = calculatePlatformCommission(grossRevenue, rule);

        // 2. 计算渠道佣金
        Long channelCommission = calculateChannelCommission(
            grossRevenue, order.getChannelType(), rule
        );

        // 3. 计算配送费
        Long deliveryFee = calculateDeliveryFee(order, rule);

        // 4. 计算支付手续费
        Long paymentFee = calculatePaymentFee(order.getPaidAmount(), rule);

        // 5. 计算商户应收
        Long merchantReceivable = grossRevenue
            - platformCommission
            - channelCommission
            - deliveryFee
            - paymentFee;

        return CommissionResult.builder()
            .platformCommission(platformCommission)
            .channelCommission(channelCommission)
            .deliveryFee(deliveryFee)
            .paymentFee(paymentFee)
            .merchantReceivable(merchantReceivable)
            .build();
    }

    private Long calculatePlatformCommission(Long grossRevenue, CommissionRule rule) {
        switch (rule.getRuleType()) {
            case PERCENTAGE:
                return Math.round(grossRevenue * rule.getBaseCommissionRate() / 100.0);
            case FIXED:
                return rule.getFixedCommissionAmount();
            case TIERED:
                return calculateTieredCommission(grossRevenue, rule.getTieredRates());
            default:
                return 0L;
        }
    }

    private Long calculateChannelCommission(
        Long grossRevenue,
        ChannelType channelType,
        CommissionRule rule
    ) {
        if (channelType == ChannelType.MEITUAN || channelType == ChannelType.ELEME) {
            return Math.round(grossRevenue * rule.getChannelCommissionRate() / 100.0);
        }
        return 0L;
    }

    private Long calculateDeliveryFee(SaleOrder order, CommissionRule rule) {
        if (order.getFulfillmentType() == FulfillmentType.THIRD_PARTY_DELIVERY) {
            // 从订单中获取配送费
            return order.getDeliveryFee();
        }
        return 0L;
    }

    private Long calculatePaymentFee(Long paidAmount, CommissionRule rule) {
        return Math.round(paidAmount * rule.getPaymentFeeRate() / 100.0);
    }

    private Long calculateTieredCommission(Long amount, List<TieredRate> tieredRates) {
        for (TieredRate rate : tieredRates) {
            if (amount >= rate.getMinAmount() && amount <= rate.getMaxAmount()) {
                return Math.round(amount * rate.getCommissionRate() / 100.0);
            }
        }
        return 0L;
    }
}
```

### 3.3 ReconciliationService（对账服务）

```java
@Service
public class ReconciliationService {

    @Autowired
    private ReconciliationTaskRepository taskRepository;

    @Autowired
    private SettlementFactRepository settlementFactRepository;

    /**
     * 执行订单对账
     */
    public ReconciliationResult executeOrderReconciliation(String taskId) {
        ReconciliationTask task = taskRepository.findById(taskId)
            .orElseThrow(() -> new ReconciliationTaskNotFoundException(taskId));

        task.start();

        try {
            // 1. 查询内部订单
            List<SettlementFact> internalFacts = settlementFactRepository
                .findByDateRange(task.getReconcileStartDate(), task.getReconcileEndDate());

            // 2. 查询外部平台订单
            List<ExternalOrder> externalOrders = queryExternalOrders(
                task.getDataSourceB(),
                task.getReconcileStartDate(),
                task.getReconcileEndDate()
            );

            // 3. 匹配订单
            List<ReconciliationDiff> diffs = matchOrders(internalFacts, externalOrders);

            // 4. 更新任务结果
            task.complete(
                internalFacts.size(),
                internalFacts.size() - diffs.size(),
                diffs.size()
            );

            // 5. 保存差异记录
            diffs.forEach(diff -> diff.setReconciliationTaskId(taskId));
            reconciliationDiffRepository.saveAll(diffs);

            taskRepository.save(task);

            return ReconciliationResult.success(task);

        } catch (Exception e) {
            task.fail(e.getMessage());
            taskRepository.save(task);
            return ReconciliationResult.failure(e.getMessage());
        }
    }

    private List<ReconciliationDiff> matchOrders(
        List<SettlementFact> internalFacts,
        List<ExternalOrder> externalOrders
    ) {
        List<ReconciliationDiff> diffs = new ArrayList<>();

        Map<String, ExternalOrder> externalMap = externalOrders.stream()
            .collect(Collectors.toMap(ExternalOrder::getOrderNo, o -> o));

        for (SettlementFact fact : internalFacts) {
            ExternalOrder external = externalMap.get(fact.getOrderId());

            if (external == null) {
                // 内部有但外部没有
                diffs.add(ReconciliationDiff.missingInExternal(fact));
            } else if (!fact.getActualSettlementAmount().equals(external.getAmount())) {
                // 金额不匹配
                diffs.add(ReconciliationDiff.amountMismatch(fact, external));
            }
        }

        return diffs;
    }
}
```
