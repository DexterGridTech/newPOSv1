### 2.2 两级缓存设计

```java
@Configuration
public class CacheConfig {

    /**
     * Caffeine 本地缓存配置
     */
    @Bean
    public Cache<String, Object> localCache() {
        return Caffeine.newBuilder()
            .maximumSize(10000)
            .expireAfterWrite(5, TimeUnit.MINUTES)
            .recordStats()
            .build();
    }

    /**
     * 两级缓存管理器
     */
    @Bean
    public CacheManager cacheManager(
        RedisConnectionFactory redisConnectionFactory,
        Cache<String, Object> localCache
    ) {
        return new TwoLevelCacheManager(redisConnectionFactory, localCache);
    }
}

/**
 * 两级缓存实现
 */
public class TwoLevelCache implements Cache {

    private final Cache<String, Object> l1Cache; // Caffeine
    private final RedisTemplate<String, Object> l2Cache; // Redis

    @Override
    public ValueWrapper get(Object key) {
        // 1. 先查L1缓存
        Object value = l1Cache.getIfPresent(key);
        if (value != null) {
            return new SimpleValueWrapper(value);
        }

        // 2. 查L2缓存
        value = l2Cache.opsForValue().get(key);
        if (value != null) {
            // 回填L1缓存
            l1Cache.put((String) key, value);
            return new SimpleValueWrapper(value);
        }

        return null;
    }

    @Override
    public void put(Object key, Object value) {
        // 同时写入L1和L2
        l1Cache.put((String) key, value);
        l2Cache.opsForValue().set((String) key, value, 1, TimeUnit.HOURS);
    }

    @Override
    public void evict(Object key) {
        // 同时清除L1和L2
        l1Cache.invalidate(key);
        l2Cache.delete((String) key);
    }
}
```

### 2.5 缓存穿透防护

```java
@Service
public class ProductCacheService {

    @Autowired
    private BloomFilter<String> productBloomFilter;

    @Autowired
    private Cache<String, ProductDetailDTO> productCache;

    @Autowired
    private ProductRepository productRepository;

    /**
     * 查询商品（防穿透）
     */
    public ProductDetailDTO getProduct(String productId) {
        // 1. 布隆过滤器检查
        if (!productBloomFilter.mightContain(productId)) {
            return null;
        }

        // 2. 查询缓存
        String cacheKey = String.format(CacheKeyConstants.PRODUCT_DETAIL, productId);
        ProductDetailDTO cached = productCache.getIfPresent(cacheKey);
        if (cached != null) {
            return cached;
        }

        // 3. 查询数据库
        Product product = productRepository.findById(productId).orElse(null);

        if (product != null) {
            ProductDetailDTO dto = ProductDetailDTO.from(product);
            productCache.put(cacheKey, dto);
            return dto;
        } else {
            // 缓存空值，防止穿透
            productCache.put(cacheKey, ProductDetailDTO.empty());
            return null;
        }
    }
}
```

```java
@Configuration
public class DataSourceConfig {

    @Bean
    @ConfigurationProperties("spring.datasource.master")
    public DataSource masterDataSource() {
        return DataSourceBuilder.create().build();
    }

    @Bean
    @ConfigurationProperties("spring.datasource.slave")
    public DataSource slaveDataSource() {

    }
}
```

### 8.1 认证与授权

**JWT Token 配置**：
```java
@Configuration
public class JwtConfig {

    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.expiration}")
    private Long expiration;

    /**
     * 生成 Token
     */
    public String generateToken(Staff staff) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("staffId", staff.getStaffId());
        claims.put("staffName", staff.getStaffName());
        claims.put("tenantId", staff.getTenantId());
        claims.put("storeId", staff.getStoreId());
        claims.put("roles", staff.getRoleIds());

        return Jwts.builder()
            .setClaims(claims)
            .setSubject(staff.getUsername())
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + expiration * 1000))
            .signWith(SignatureAlgorithm.HS512, secret)
            .compact();
    }

    /**
     * 验证 Token
     */
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}
```

**权限拦截器**：
```java
@Component
public class AuthorizationInterceptor implements HandlerInterceptor {

    @Autowired
    private JwtConfig jwtConfig;

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) {
        // 1. 获取 Token
        String token = request.getHeader("Authorization");
        if (token == null || !token.startsWith("Bearer ")) {
            throw new UnauthorizedException("未登录");
        }

        token = token.substring(7);

        // 2. 验证 Token
        if (!jwtConfig.validateToken(token)) {
            throw new UnauthorizedException("Token无效");
        }

        // 3. 提取用户信息
        Claims claims = jwtConfig.getClaimsFromToken(token);
        String staffId = claims.get("staffId", String.class);
        String tenantId = claims.get("tenantId", String.class);

        // 4. 设置上下文
        SecurityContextHolder.setStaffId(staffId);
        SecurityContextHolder.setTenantId(tenantId);

        return true;
    }
}
```

### 8.2 数据安全

**敏感数据加密**：
```java
@Component
public class DataEncryptionService {

    private static final String ALGORITHM = "AES";
    private static final String TRANSFORMATION = "AES/CBC/PKCS5Padding";

    @Value("${encryption.key}")
    private String encryptionKey;

    /**
     * 加密
     */
    public String encrypt(String plainText) {
        try {
            SecretKeySpec keySpec = new SecretKeySpec(encryptionKey.getBytes(), ALGORITHM);
            Cipher cipher = Cipher.getInstance(TRANSFORMATION);
            cipher.init(Cipher.ENCRYPT_MODE, keySpec);
            byte[] encrypted = cipher.doFinal(plainText.getBytes());
            return Base64.getEncoder().encodeToString(encrypted);
        } catch (Exception e) {
            throw new EncryptionException("加密失败", e);
        }
    }

    /**
     * 解密
     */
    public String decrypt(String encryptedText) {
        try {
            SecretKeySpec keySpec = new SecretKeySpec(encryptionKey.getBytes(), ALGORITHM);
            Cipher cipher = Cipher.getInstance(TRANSFORMATION);
            cipher.init(Cipher.DECRYPT_MODE, keySpec);
            byte[] decrypted = cipher.doFinal(Base64.getDecoder().decode(encryptedText));
            return new String(decrypted);
        } catch (Exception e) {
            throw new EncryptionException("解密失败", e);
        }
    }
}
```

## 11. SLA / RTO / RPO 定义

### 11.1 服务可用性目标（SLA）

| 服务等级 | 服务 | 可用性目标 | 允许月停机时长 |
|---|---|---|---|
| P0（核心交易） | 交易服务、支付回调、渠道订单接收 | 99.95% | ≤22分钟/月 |
| P1（核心履约） | 履约服务、TDP终端推送、打印服务 | 99.9% | ≤44分钟/月 |
| P2（经营支撑） | 商品服务、菜单查询、结算服务 | 99.5% | ≤3.6小时/月 |
| P3（后台管理） | 运营后台、报表查询、对账任务 | 99% | ≤7.2小时/月 |

### 11.2 恢复时间目标（RTO）

| 故障类型 | RTO目标 | 恢复手段 |
|---|---|---|
| 单实例宕机 | < 30秒 | K8s自动重启 + 健康检查 |
| 数据库主库故障 | < 60秒 | Patroni自动主从切换 |
| Redis主节点故障 | < 30秒 | Redis Sentinel自动切换 |
| 整个可用区故障 | < 5分钟 | 流量切换到备用可用区 |
| 数据中心级故障（P2） | < 30分钟 | 异地容灾切换（手动触发） |

### 11.3 恢复点目标（RPO）

| 数据类型 | RPO目标 | 备份策略 |
|---|---|---|
| 订单/支付数据 | 0（零丢失） | 同步主从复制 + WAL归档 |
| 商品/菜单数据 | < 5分钟 | 异步主从复制 |
| 日志/审计数据 | < 1小时 | 定时备份到对象存储 |
| 监控指标数据 | < 24小时 | Prometheus TSDB定时快照 |

### 11.4 关键接口性能目标

| 接口 | P99延迟 | P999延迟 | 备注 |
|---|---|---|---|
| 下单接口 | < 500ms | < 1s | 含商品快照+库存锁定 |
| 支付回调处理 | < 200ms | < 500ms | 必须快速响应支付渠道 |
| 菜单查询 | < 50ms | < 100ms | 走缓存 |
| 渠道订单接收 | < 300ms | < 800ms | 含签名验证+幂等检查 |
| KDS工作单推送 | < 100ms | < 300ms | 基于TDP推送 |

### 11.5 灾难恢复操作手册

**场景一：数据库主库不可用**
```bash
# 1. 确认主库状态
patronictl -c /etc/patroni/config.yml list

# 2. 若Patroni未自动切换，手动触发
patronictl -c /etc/patroni/config.yml failover restaurant-cluster --master <old-master>

# 3. 验证新主库可写
psql -h <new-master-ip> -U postgres -c "SELECT pg_is_in_recovery();"
# 应返回 f（false）

# 4. 更新应用配置（若使用静态IP）
kubectl set env deployment/transaction-service DB_MASTER_HOST=<new-master-ip>
```

**场景二：Redis集群不可用（缓存雪崩）**
```bash
# 1. 开启本地缓存降级模式（Caffeine兜底）
kubectl set env deployment/product-service CACHE_FALLBACK_MODE=LOCAL_ONLY

# 2. 限流保护数据库（临时降低QPS上限）
kubectl set env deployment/transaction-service DB_MAX_CONNECTIONS=50

# 3. 重建Redis集群后，预热缓存
curl -X POST http://product-service/internal/cache/warmup
```

**场景三：消息队列积压（RocketMQ）**
```bash
# 查看积压量
mqadmin consumerProgress -n <nameserver> -g <consumer-group>

# 若积压超过10万条，临时扩容消费者
kubectl scale deployment fulfillment-service --replicas=10

# 若消息已过期（超过保留时长），跳过并告警
mqadmin resetOffsetByTime -n <nameserver> -g <consumer-group> -t <topic> -s now
```
