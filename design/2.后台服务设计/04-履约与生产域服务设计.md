# 04-履约与生产域服务设计

## 1. 服务概述

履约与生产域服务（fulfillment-service）负责订单履约、生产计划、工作单派发、交付协调等功能。

**核心职责**：
- 履约订单管理
- 生产计划生成
- 工作单派发与执行
- 交付任务协调
- 配送任务管理
- 打印任务管理
- 叫号服务管理

**核心聚合根**：
- FulfillmentOrder（履约订单）
- ProductionPlan（生产计划）
- HandoffTask（交接任务）
- DeliveryTask（配送任务）

### 3.2 FulfillmentOrder 聚合根

```java
@Entity
@Table(name = "fulfillment_orders")
public class FulfillmentOrder {

    @Id
    private String fulfillmentOrderId;

    private String orderId;
    private String storeId;

    @Enumerated(EnumType.STRING)
    private FulfillmentType fulfillmentType;

    @Enumerated(EnumType.STRING)
    private OrderScene orderScene;

    @Enumerated(EnumType.STRING)
    private Priority priority;

    private String customerName;
    private String customerPhoneSuffix;

    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private ServiceContext serviceContext;

    @OneToMany(mappedBy = "fulfillmentOrder", cascade = CascadeType.ALL)
    private List<FulfillmentOrderItem> items;

    @Enumerated(EnumType.STRING)
    private FulfillmentStatus status;

    @Enumerated(EnumType.STRING)
    private ProductionStatus productionStatus;

    @Enumerated(EnumType.STRING)
    private HandoffStatus handoffStatus;

    private LocalDateTime createdAt;
    private LocalDateTime acceptedAt;
    private LocalDateTime productionStartedAt;
    private LocalDateTime productionCompletedAt;
    private LocalDateTime handoffCompletedAt;

    private LocalDateTime expectedReadyTime;
    private LocalDateTime expectedDeliveryTime;

    private Integer productionSlaSec;
    private Integer handoffSlaSec;
    private Boolean isOverdue;
    private Integer overdueSeconds;

    @Version
    private Integer version;

    /**
     * 创建履约订单
     */
    public static FulfillmentOrder create(CreateFulfillmentOrderCommand command) {
        FulfillmentOrder order = new FulfillmentOrder();
        order.fulfillmentOrderId = IdGenerator.generateFulfillmentOrderId();
        order.orderId = command.getOrderId();
        order.storeId = command.getStoreId();
        order.fulfillmentType = command.getFulfillmentType();
        order.orderScene = command.getOrderScene();
        order.priority = calculatePriority(command);
        order.customerName = command.getCustomerName();
        order.customerPhoneSuffix = command.getCustomerPhoneSuffix();
        order.serviceContext = command.getServiceContext();
        order.status = FulfillmentStatus.PLAN_PENDING;
        order.productionStatus = ProductionStatus.NOT_STARTED;
        order.handoffStatus = HandoffStatus.PENDING;
        order.createdAt = LocalDateTime.now();
        order.acceptedAt = command.getAcceptedAt();

        order.items = command.getItems().stream()
            .map(item -> FulfillmentOrderItem.create(order, item))
            .collect(Collectors.toList());

        order.addDomainEvent(new FulfillmentOrderCreatedEvent(order));

        return order;
    }

    /**
     * 生成生产计划
     */
    public void planProduction(ProductionPlan plan) {
        if (this.status != FulfillmentStatus.PLAN_PENDING) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.PRODUCTION_PLANNED;
        this.expectedReadyTime = plan.getEstimatedCompletionTime();

        this.addDomainEvent(new ProductionPlanGeneratedEvent(this, plan));
    }

    /**
     * 开始生产
     */
    public void startProduction() {
        if (this.status != FulfillmentStatus.PRODUCTION_PLANNED) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.IN_PRODUCTION;
        this.productionStatus = ProductionStatus.IN_PROGRESS;
        this.productionStartedAt = LocalDateTime.now();

        this.addDomainEvent(new ProductionStartedEvent(this));
    }

    /**
     * 生产完成
     */
    public void completeProduction() {
        if (this.status != FulfillmentStatus.IN_PRODUCTION) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.PRODUCTION_DONE;
        this.productionStatus = ProductionStatus.COMPLETED;
        this.productionCompletedAt = LocalDateTime.now();

        checkSLA();

        this.addDomainEvent(new ProductionCompletedEvent(this));
    }

    /**
     * 准备交付
     */
    public void prepareHandoff() {
        if (this.status != FulfillmentStatus.PRODUCTION_DONE) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.HANDOFF_READY;
        this.handoffStatus = HandoffStatus.READY;

        this.addDomainEvent(new HandoffReadyEvent(this));
    }

    /**
     * 开始交付
     */
    public void startHandoff() {
        if (this.status != FulfillmentStatus.HANDOFF_READY) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.IN_HANDOFF;
        this.handoffStatus = HandoffStatus.IN_PROGRESS;

        this.addDomainEvent(new HandoffStartedEvent(this));
    }

    /**
     * 完成交付
     */
    public void completeHandoff() {
        if (this.status != FulfillmentStatus.IN_HANDOFF) {
            throw new IllegalStateException("履约订单状态不正确");
        }

        this.status = FulfillmentStatus.FULFILLED;
        this.handoffStatus = HandoffStatus.COMPLETED;
        this.handoffCompletedAt = LocalDateTime.now();

        this.addDomainEvent(new FulfillmentCompletedEvent(this));
    }

    private void checkSLA() {
        if (this.productionSlaSec != null && this.productionStartedAt != null) {
            long actualSeconds = Duration.between(
                this.productionStartedAt,
                this.productionCompletedAt
            ).getSeconds();

            if (actualSeconds > this.productionSlaSec) {
                this.isOverdue = true;
                this.overdueSeconds = (int) (actualSeconds - this.productionSlaSec);
            }
        }
    }

    private static Priority calculatePriority(CreateFulfillmentOrderCommand command) {
        if (command.getFulfillmentType() == FulfillmentType.THIRD_PARTY_DELIVERY ||
            command.getFulfillmentType() == FulfillmentType.PLATFORM_DELIVERY) {
            return Priority.HIGH;
        }
        return Priority.NORMAL;
    }
}
```

### 3.3 ProductionPlan 聚合根

```java
@Entity
@Table(name = "production_plans")
public class ProductionPlan {

    @Id
    private String productionPlanId;

    private String fulfillmentOrderId;
    private String storeId;

    @Enumerated(EnumType.STRING)
    private PlanType planType;

    @Enumerated(EnumType.STRING)
    private PlanStatus planStatus;

    private Integer totalWorkUnits;
    private Integer completedWorkUnits;

    private Integer estimatedDuration;
    private Integer actualDuration;

    @OneToMany(mappedBy = "productionPlan", cascade = CascadeType.ALL)
    private List<WorkUnit> workUnits;

    private LocalDateTime createdAt;
    private LocalDateTime startedAt;
    private LocalDateTime completedAt;

    /**
     * 创建生产计划
     */
    public static ProductionPlan create(
        FulfillmentOrder order,
        List<WorkUnit> workUnits
    ) {
        ProductionPlan plan = new ProductionPlan();
        plan.productionPlanId = IdGenerator.generateProductionPlanId();
        plan.fulfillmentOrderId = order.getFulfillmentOrderId();
        plan.storeId = order.getStoreId();
        plan.planType = PlanType.IMMEDIATE;
        plan.planStatus = PlanStatus.PENDING;
        plan.totalWorkUnits = workUnits.size();
        plan.completedWorkUnits = 0;
        plan.workUnits = workUnits;
        plan.createdAt = LocalDateTime.now();

        // 计算预计时长（取所有工作单中最长的预计时长，并行执行）
        plan.estimatedDuration = workUnits.stream()
            .mapToInt(WorkUnit::getEstimatedDuration)
            .max()
            .orElse(0);

        return plan;
    }

    /**
     * 派发工作单
     */
    public void dispatchWorkUnits() {
        if (this.planStatus != PlanStatus.PENDING) {
            throw new IllegalStateException("生产计划状态不正确");
        }

        this.planStatus = PlanStatus.ACTIVE;
        this.startedAt = LocalDateTime.now();

        for (WorkUnit workUnit : workUnits) {
            workUnit.dispatch();
        }

        this.addDomainEvent(new WorkUnitsDispatchedEvent(this));
    }

    /**
     * 工作单完成回调
     */
    public void onWorkUnitCompleted(String workUnitId) {
        this.completedWorkUnits++;

        if (this.completedWorkUnits >= this.totalWorkUnits) {
            this.planStatus = PlanStatus.COMPLETED;
            this.completedAt = LocalDateTime.now();
            this.actualDuration = (int) Duration.between(
                this.startedAt,
                this.completedAt
            ).getSeconds();

            this.addDomainEvent(new AllWorkUnitsCompletedEvent(this));
        }
    }

    public LocalDateTime getEstimatedCompletionTime() {
        return this.createdAt.plusSeconds(this.estimatedDuration);
    }
}
```

## 5. 应用服务设计

### 5.1 FulfillmentApplicationService

```java
@Service
@Transactional
public class FulfillmentApplicationService {

    @Autowired
    private FulfillmentOrderRepository fulfillmentOrderRepository;

    @Autowired
    private WorkUnitRepository workUnitRepository;

    @Autowired
    private ProductionPlanService productionPlanService;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 创建履约订单
     */
    public CreateFulfillmentOrderResponse createFulfillmentOrder(
        CreateFulfillmentOrderCommand command
    ) {
        // 创建履约订单
        FulfillmentOrder order = FulfillmentOrder.create(command);

        // 保存订单
        fulfillmentOrderRepository.save(order);

        // 发布事件
        eventPublisher.publish(order.getDomainEvents());

        // 异步生成生产计划
        asyncGenerateProductionPlan(order.getFulfillmentOrderId());

        return CreateFulfillmentOrderResponse.builder()
            .fulfillmentOrderId(order.getFulfillmentOrderId())
            .build();
    }

    /**
     * 生成生产计划
     */
    public void generateProductionPlan(String fulfillmentOrderId) {
        FulfillmentOrder order = fulfillmentOrderRepository.findById(fulfillmentOrderId)
            .orElseThrow(() -> new FulfillmentOrderNotFoundException(fulfillmentOrderId));

        // 调用生产计划服务生成计划
        ProductionPlan plan = productionPlanService.generatePlan(order);

        // 更新履约订单状态
        order.planProduction(plan);

        fulfillmentOrderRepository.save(order);
        eventPublisher.publish(order.getDomainEvents());

        // 派发工作单
        plan.dispatchWorkUnits();
        eventPublisher.publish(plan.getDomainEvents());
    }

    /**
     * 工作单开始制作
     */
    public void startWorkUnit(String workUnitId, String operatorId) {
        WorkUnit workUnit = workUnitRepository.findById(workUnitId)
            .orElseThrow(() -> new WorkUnitNotFoundException(workUnitId));

        workUnit.start(operatorId);
        workUnitRepository.save(workUnit);

        // 检查是否是该履约订单的第一个开始的工作单
        ProductionPlan plan = workUnit.getProductionPlan();
        FulfillmentOrder order = fulfillmentOrderRepository
            .findById(plan.getFulfillmentOrderId())
            .orElseThrow();

        if (order.getStatus() == FulfillmentStatus.PRODUCTION_PLANNED) {
            order.startProduction();
            fulfillmentOrderRepository.save(order);
            eventPublisher.publish(order.getDomainEvents());
        }
    }

    /**
     * 工作单完成
     */
    public void completeWorkUnit(String workUnitId) {
        WorkUnit workUnit = workUnitRepository.findById(workUnitId)
            .orElseThrow(() -> new WorkUnitNotFoundException(workUnitId));

        workUnit.complete();
        workUnitRepository.save(workUnit);

        // 通知生产计划
        ProductionPlan plan = workUnit.getProductionPlan();
        plan.onWorkUnitCompleted(workUnitId);

        if (plan.getPlanStatus() == PlanStatus.COMPLETED) {
            // 更新履约订单
            FulfillmentOrder order = fulfillmentOrderRepository
                .findById(plan.getFulfillmentOrderId())
                .orElseThrow();

            order.completeProduction();
            fulfillmentOrderRepository.save(order);

            eventPublisher.publish(order.getDomainEvents());
        }
    }

    /**
     * 准备交付
     */
    public void prepareHandoff(String fulfillmentOrderId) {
        FulfillmentOrder order = fulfillmentOrderRepository.findById(fulfillmentOrderId)
            .orElseThrow(() -> new FulfillmentOrderNotFoundException(fulfillmentOrderId));

        order.prepareHandoff();

        fulfillmentOrderRepository.save(order);
        eventPublisher.publish(order.getDomainEvents());
    }

    /**
     * 完成交付
     */
    public void completeHandoff(String fulfillmentOrderId) {
        FulfillmentOrder order = fulfillmentOrderRepository.findById(fulfillmentOrderId)
            .orElseThrow(() -> new FulfillmentOrderNotFoundException(fulfillmentOrderId));

        order.startHandoff();
        order.completeHandoff();

        fulfillmentOrderRepository.save(order);
        eventPublisher.publish(order.getDomainEvents());
    }

    @Async
    private void asyncGenerateProductionPlan(String fulfillmentOrderId) {

            Thread.sleep(100); // 短暂延迟，确保事务提交
            generateProductionPlan(fulfillmentOrderId);
        } catch (Exception e) {
            log.error("生成生产计划失败: {}", fulfillmentOrderId, e);
```

## 6. 生产计划生成算法实现

### 6.1 ProductionPlanService

```java
@Service
public class ProductionPlanService {

    @Autowired
    private ProductServiceClient productServiceClient;

    /**
     * 生成生产计划
     *
     * 算法流程：
     * 1. 提取订单行
     * 2. 查询每个商品的生产画像（包含工作站类型和预计时长）
     * 3. 按工作站类型分组（同一工作站的商品归入同一工作单）
     * 4. 为每个工作站创建工作单
     * 5. 如果是外卖订单，额外生成打包工作单
     * 6. 组装生产计划
     */
    public ProductionPlan generatePlan(FulfillmentOrder order) {
        // 1. 提取订单行
        List<FulfillmentOrderItem> items = order.getItems();

        // 2. 查询商品生产画像
        Map<String, ProductionProfile> profiles = getProductionProfiles(items);

        // 3. 按工作站分组
        Map<StationType, List<WorkUnitItem>> groupedItems = groupByStation(items, profiles);

        // 4. 生成工作单
        List<WorkUnit> workUnits = new ArrayList<>();
        int sequence = 1;

        for (Map.Entry<StationType, List<WorkUnitItem>> entry : groupedItems.entrySet()) {
            WorkUnit workUnit = createWorkUnit(
                order,
                entry.getKey(),
                entry.getValue(),
                sequence++
            );
            workUnits.add(workUnit);
        }

        // 5. 外卖需要打包工作单
        if (order.getFulfillmentType() == FulfillmentType.THIRD_PARTY_DELIVERY ||
            order.getFulfillmentType() == FulfillmentType.PLATFORM_DELIVERY) {
            WorkUnit packageUnit = createPackageWorkUnit(order, sequence);
            workUnits.add(packageUnit);
        }

        // 6. 创建生产计划
        return ProductionPlan.create(order, workUnits);
    }

    /**
     * 查询商品生产画像
     */
    private Map<String, ProductionProfile> getProductionProfiles(
        List<FulfillmentOrderItem> items
    ) {
        List<String> productIds = items.stream()
            .map(FulfillmentOrderItem::getProductId)
            .distinct()
            .collect(Collectors.toList());

        return productServiceClient.getProductionProfiles(productIds);
    }

    /**
     * 按工作站分组
     * 将同一工作站类型的商品聚合在一起，形成一个工作单
     */
    private Map<StationType, List<WorkUnitItem>> groupByStation(
        List<FulfillmentOrderItem> items,
        Map<String, ProductionProfile> profiles
    ) {
        Map<StationType, List<WorkUnitItem>> grouped = new HashMap<>();

        for (FulfillmentOrderItem item : items) {
            ProductionProfile profile = profiles.get(item.getProductId());
            StationType stationType = profile.getStationType();

            WorkUnitItem workItem = WorkUnitItem.builder()
                .productId(item.getProductId())
                .productName(item.getProductSnapshot().getProductName())
                .variantName(item.getProductSnapshot().getSelectedVariant().getVariantName())
                .modifiers(item.getProductSnapshot().getSelectedModifiers())
                .quantity(item.getQuantity())
                .specifications(item.getRemark())
                .build();

            grouped.computeIfAbsent(stationType, k -> new ArrayList<>()).add(workItem);
        }

        return grouped;
    }

    /**
     * 创建生产工作单
     */
    private WorkUnit createWorkUnit(
        FulfillmentOrder order,
        StationType stationType,
        List<WorkUnitItem> items,
        int sequence
    ) {
        WorkUnit workUnit = new WorkUnit();
        workUnit.setWorkUnitId(IdGenerator.generateWorkUnitId());
        workUnit.setStationType(stationType);
        workUnit.setStationName(stationType.getDisplayName());
        workUnit.setWorkUnitType(WorkUnitType.PRODUCTION);
        workUnit.setPriority(order.getPriority());
        workUnit.setSequence(sequence);
        workUnit.setStatus(WorkUnitStatus.PENDING);
        workUnit.setItems(items);

        // 计算预计时长：每个商品假设60秒，后续可根据生产画像精细化
        int estimatedDuration = items.stream()
            .mapToInt(item -> item.getQuantity() * 60) // 假设每个商品60秒
            .sum();
        workUnit.setEstimatedDuration(estimatedDuration);

        return workUnit;
    }

    /**
     * 创建打包工作单
     */
    private WorkUnit createPackageWorkUnit(FulfillmentOrder order, int sequence) {
        WorkUnit workUnit = new WorkUnit();
        workUnit.setWorkUnitId(IdGenerator.generateWorkUnitId());
        workUnit.setStationType(StationType.PACKAGE);
        workUnit.setStationName("打包工作站");
        workUnit.setWorkUnitType(WorkUnitType.PACKAGE);
        workUnit.setPriority(order.getPriority());
        workUnit.setSequence(sequence);
        workUnit.setStatus(WorkUnitStatus.PENDING);
        workUnit.setEstimatedDuration(120); // 打包预计2分钟

        return workUnit;
    }
}
```

### 7.2 TDP Write Client

```java
@Service
public class TDPWriteClient {

    @Autowired
    private RestTemplate restTemplate;

    @Value("${tdp.service.url}")
    private String tdpServiceUrl;

    @Autowired
    private RetryQueue retryQueue;

    /**
     * 写入 Projection 到 TDP
     */
    public void writeProjection(ProjectionWriteRequest request) {
        String url = tdpServiceUrl + "/v1/projections";

        try {
            restTemplate.postForEntity(url, request, Void.class);
        } catch (Exception e) {
            log.error("写入 TDP Projection 失败: topic={}, itemKey={}",
                request.getTopic(), request.getItemKey(), e);
            // 写入失败队列，异步重试
            retryQueue.add(request);
        }
    }
}

@Data
@Builder
public class ProjectionWriteRequest {
    private String tenantId;
    private String topic;
    private String itemKey;
    private String scopeType;
    private String scopeId;
    private String operation;  // upsert / delete
    private Map<String, Object> payload;
    private String sourceSystem;
    private Integer ttlSeconds;
}
```

```java
@Service
@Transactional
public class DeliveryScheduleService {

    @Autowired
    private FulfillmentOrderRepository fulfillmentOrderRepository;

    @Autowired
    private ChannelStatusSyncService channelStatusSyncService;

     * 通知骑手取餐（外卖订单生产完成后触发）

            .findById(event.getFulfillmentOrderId())

                getChannelType(order),
                OrderStatus.PRODUCTION_COMPLETED
            );
        } else if (order.getFulfillmentType() == FulfillmentType.PLATFORM_DELIVERY) {
            // 平台自有配送，创建配送任务
            createDeliveryTask(order);
        }
    }

    /**
     * 创建配送任务
     */
    private void createDeliveryTask(FulfillmentOrder order) {
        DeliveryTask task = DeliveryTask.builder()
            .taskId(IdGenerator.generateDeliveryTaskId())
            .fulfillmentOrderId(order.getFulfillmentOrderId())
            .storeId(order.getStoreId())
            .deliveryAddress(order.getServiceContext().getDeliveryAddress())
            .deliveryContact(order.getServiceContext().getDeliveryContact())
            .deliveryPhone(order.getServiceContext().getDeliveryPhone())
            .status(DeliveryStatus.PENDING_ASSIGN)
            .createdAt(LocalDateTime.now())
            .build();

        deliveryTaskRepository.save(task);
        log.info("配送任务已创建: taskId={}, fulfillmentOrderId={}",
            task.getTaskId(), order.getFulfillmentOrderId());
    }

    private ChannelType getChannelType(FulfillmentOrder order) {
        // 从订单关联的 channel_type 字段读取，不得根据 orderScene 猜测渠道类型
        // 原因：DELIVERY_PLATFORM 场景可能来自美团、饿了么、抖音等多个渠道，
        // 必须通过订单真相源（transaction-service）传入的 channelType 字段确定
        if (order.getChannelType() != null) {
            return order.getChannelType();
        }
        // 无渠道类型时返回 null，由调用方决定是否需要回传
        return null;
    }
```

```java
        deviceCommandService.sendCommand(deviceId, CommandType.PRINT, payload,
            "print-" + task.getTaskId());

        task.setStatus(PrintStatus.SENT);
        task.setSentAt(LocalDateTime.now());
        printTaskRepository.save(task);

        return task;
    }

    /**
     * 监听生产完成事件，自动触发打印
     */
    @EventListener
    public void onProductionCompleted(ProductionCompletedEvent event) {
        // 查询门店打印机配置，自动创建出品口小票
        List<Device> printers = deviceRepository.findByStoreIdAndType(
            event.getStoreId(), DeviceType.PRINTER);
        for (Device printer : printers) {
            createPrintTask(event.getOrderId(), event.getStoreId(),
                ReceiptType.KITCHEN_GATE, printer.getDeviceId(), TriggerType.AUTO);
        }
    }

    /**
     * 手动重打
     */
    public PrintTask reprintTask(String originalTaskId) {
        PrintTask original = printTaskRepository.findById(originalTaskId)
            .orElseThrow(() -> new NotFoundException("打印任务不存在"));

        return createPrintTask(
            original.getOrderId(),
            original.getStoreId(),
            original.getReceiptType(),
            original.getTargetDeviceId(),
            TriggerType.MANUAL
        );
    }

    private String buildPrintPayload(PrintTask task) {
        // 构建打印内容 JSON（根据 ReceiptType 生成不同格式）
        return JSON.toJSONString(Map.of(
            "taskId", task.getTaskId(),
            "orderId", task.getOrderId(),
            "receiptType", task.getReceiptType(),
            "timestamp", LocalDateTime.now().toString()
        ));
    }
```

```java
@Service
@Transactional
public class PickupCallService {

    @Autowired
    private PickupCallRepository pickupCallRepository;

    @Autowired
    private ApplicationEventPublisher eventPublisher;

    /**
     * 触发叫号（生产完成后自动调用）
     */
    public PickupCall triggerCall(String orderId, String storeId, String pickupCode) {
        PickupCall call = new PickupCall();
        call.setCallId(UUID.randomUUID().toString());
        call.setOrderId(orderId);
        call.setStoreId(storeId);
        call.setPickupCode(pickupCode);
        call.setCallStatus(CallStatus.CALLED);
        call.setCalledAt(LocalDateTime.now());
        call.setCallCount(1);
        call.setExpiredAt(LocalDateTime.now().plusMinutes(30));
        pickupCallRepository.save(call);

        eventPublisher.publishEvent(new PickupCallTriggeredEvent(call));
        return call;
    }

    /**
     * 重复叫号（定时任务：叫号后N分钟未取餐）
     */
    @Scheduled(fixedDelay = 60000)
    public void retryUnpickedCalls() {
        List<PickupCall> pending = pickupCallRepository
            .findCalledButNotPickedUp(LocalDateTime.now().minusMinutes(5));

        for (PickupCall call : pending) {
            if (call.getCallCount() < 3) {
                call.setCallCount(call.getCallCount() + 1);
                call.setCalledAt(LocalDateTime.now());
                pickupCallRepository.save(call);
                eventPublisher.publishEvent(new PickupCallTriggeredEvent(call));
            } else {
                call.setCallStatus(CallStatus.EXPIRED);
                pickupCallRepository.save(call);
            }
        }
    }

    /**
     * 确认取餐
     */
    public void confirmPickup(String callId, String operatorId) {
        PickupCall call = pickupCallRepository.findById(callId)
            .orElseThrow(() -> new NotFoundException("叫号记录不存在"));
        call.setCallStatus(CallStatus.PICKED_UP);
        call.setPickedUpAt(LocalDateTime.now());
        pickupCallRepository.save(call);
    }

    /**
     * 监听生产完成事件，自动触发叫号（仅自取场景）
     */
    @EventListener
    public void onProductionCompleted(ProductionCompletedEvent event) {
        if (event.getFulfillmentType().equals("SELF_PICKUP")) {
            triggerCall(event.getOrderId(), event.getStoreId(), event.getPickupCode());
        }
    }
```

### 11.1 领域事件定义

> **内外部事件区分说明：**
>
> | 事件类 | 类型 | 传输机制 | 消费方 |
> |--------|------|----------|--------|
> | `ProductionCompletedEvent` | **外部事件** | RocketMQ | 渠道集成域、交易域 |
> | `FulfillmentCompletedEvent` | **外部事件** | RocketMQ | 交易域（更新订单状态）、结算域 |
> | `WorkUnitDispatchedEvent` | **内部事件** | Spring ApplicationEvent | 同服务内 WebSocket 推送（KDS） |
> | `WorkUnitCompletedEvent` | **内部事件** | Spring ApplicationEvent | 同服务内 WebSocket 推送（KDS） |
> | `WorkUnitsDispatchedEvent` | **内部事件** | Spring ApplicationEvent | 同服务内，驱动 ProductionPlan 状态变更 |
> | `AllWorkUnitsCompletedEvent` | **内部事件** | Spring ApplicationEvent | 同服务内，驱动 FulfillmentOrder 进入 PRODUCTION_DONE 状态 |
> | `PickupCallTriggeredEvent` | **内部事件** | Spring ApplicationEvent | 同服务内 WebSocket 推送（叫号屏） |

```java
            event.getFulfillmentOrderId());

        }
```

```java
@Service
public class WorkUnitBatchDispatchService {

    @Autowired
    private WorkUnitRepository workUnitRepository;

    @Autowired
    private DeviceCommandService deviceClient;

    /**
     * 批量派发工作单
     */
    public void batchDispatchWorkUnits(List<String> workUnitIds) {
        List<WorkUnit> workUnits = workUnitRepository.findAllById(workUnitIds);

        // 按设备分组
        Map<String, List<WorkUnit>> groupedByDevice = workUnits.stream()
            .filter(wu -> wu.getDeviceId() != null)
            .collect(Collectors.groupingBy(WorkUnit::getDeviceId));

        // 批量推送到设备
        for (Map.Entry<String, List<WorkUnit>> entry : groupedByDevice.entrySet()) {
            String deviceId = entry.getKey();
            List<WorkUnit> units = entry.getValue();

            deviceClient.batchPushWorkUnits(deviceId, units);
        }
    }
}
```

## 13. KDS设备离线降级处理

### 13.1 设备状态检测

```java
@Component
public class DeviceHeartbeatMonitor {

    // 设备心跳超时阈值：30秒无心跳视为离线
    private static final long OFFLINE_THRESHOLD_MS = 30_000;

    public DeviceStatus getDeviceStatus(String deviceId) {
        String lastHeartbeat = redisTemplate.opsForValue().get("device:heartbeat:" + deviceId);
        if (lastHeartbeat == null) return DeviceStatus.UNKNOWN;
        long elapsed = System.currentTimeMillis() - Long.parseLong(lastHeartbeat);
        return elapsed > OFFLINE_THRESHOLD_MS ? DeviceStatus.OFFLINE : DeviceStatus.ONLINE;
    }
}
```

### 13.2 工作单推送降级策略

```
KDS在线  → WebSocket实时推送工作单
KDS离线  → 工作单写入待推送队列（work_unit.dispatch_status = PENDING_DEVICE）
         → 告警通知店长（钉钉/短信）
         → 自动降级：同时触发打印机备份打印（若打印机在线）
KDS恢复  → 自动重推队列中所有 PENDING_DEVICE 的工作单
         → 按原始优先级排序，不重复推送已完成的工作单
```

```java
public void dispatchWorkUnit(WorkUnit workUnit) {
    String deviceId = workUnit.getTargetDeviceId();
    DeviceStatus status = deviceHeartbeatMonitor.getDeviceStatus(deviceId);

    if (status == DeviceStatus.ONLINE) {
        kdsWebSocketService.push(deviceId, workUnit);
        workUnit.markDispatched();
    } else {
        // 降级：入队 + 备份打印
        workUnit.markPendingDevice();
        workUnitRepository.save(workUnit);
        fallbackPrintService.printIfAvailable(workUnit); // 打印机备份
        alertService.notifyStoreManager(deviceId, "KDS设备离线，订单已备份打印");
    }
}
```

### 13.3 设备恢复后的重推

```java
@EventListener
public void onDeviceOnline(DeviceOnlineEvent event) {
    List<WorkUnit> pending = workUnitRepository
        .findPendingByDevice(event.getDeviceId());

    // 过滤掉已被人工标记完成的工作单
    pending.stream()
        .filter(wu -> wu.getStatus() != WorkUnitStatus.COMPLETED)
        .sorted(Comparator.comparing(WorkUnit::getPriority).reversed()
            .thenComparing(WorkUnit::getCreatedAt))
        .forEach(wu -> {
            kdsWebSocketService.push(event.getDeviceId(), wu);
            wu.markDispatched();
        });

    workUnitRepository.saveAll(pending);
    log.info("设备恢复，重推工作单 deviceId={} count={}", event.getDeviceId(), pending.size());
}
```
