# 04-热更新服务设计

## 1. 热更新概述

热更新服务负责终端应用的版本管理和升级发布，包括：
- 应用版本管理
- 热更新包管理
- 灰度发布策略
- 版本回滚机制

### 1.1 升级发布流程事务边界

```
【事务边界1：创建发布单】
1. 创建 UpgradeRelease
   - 发布 UpgradeReleaseCreatedEvent 到 outbox 表
   - 提交事务

【事务边界2：创建任务实例】
2. 查询目标终端列表
3. 应用灰度策略
4. 批量创建 UpgradeTaskInstance
   - 更新 UpgradeRelease.totalCount
   - 提交事务

【异步操作：委托 TDP 投递】
5. 分批调用 TDP 内部接口
   - 更新 UpgradeTaskInstance.deliveryStatus
   - 失败重试（最多3次）

【补偿操作】
- 如果发布失败，执行补偿：
  - 取消发布单
  - 取消所有待执行的任务实例
  - 通知 TDP 取消投递
  - 详见：12-分布式事务与一致性设计.md
```

```java
@Entity
@Table(name = "app_versions")
public class AppVersion {

    @Id
    private String versionId;  // 版本唯一标识

    private String appName;  // 应用名称，如 mixc-retail-pos
    private String version;  // 语义化版本号，如 2.3.1
    private String bundleVersion;  // 构建版本号，如 2026.04.06.001

    // 版本信息
    private String platform;  // 平台标识：android-rn84, electron, ios
    private String profileId;  // 设备能力模板ID，限定适用的设备类型

    // 更新包信息
    private String packageUrl;  // 更新包下载地址（对象存储URL）
    private String packageHash;  // 更新包SHA256哈希值，用于完整性校验
    private Long packageSize;  // 更新包大小（字节）

    @Enumerated(EnumType.STRING)
    private VersionType versionType;  // 版本类型：FULL_PACKAGE（完整包）, HOT_UPDATE（热更新包）

    @Enumerated(EnumType.STRING)
    private VersionStatus status;  // 版本状态：DRAFT（草稿）, PUBLISHED（已发布）, DEPRECATED（已废弃）

    // 变更说明
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private List<String> changeLog;  // 版本变更日志，用于展示给用户

    // 强制更新
    private Boolean forceUpdate;  // 是否强制更新，true时终端必须立即更新
    private String minCompatibleVersion;  // 最低兼容版本，低于此版本的终端必须更新

    // 版本依赖（P1-2）
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private VersionDependency dependency;  // 版本依赖关系，如依赖的原生壳版本

    private String createdBy;  // 创建人
    private LocalDateTime createdAt;  // 创建时间
    private LocalDateTime publishedAt;  // 发布时间
    private LocalDateTime deprecatedAt;  // 废弃时间

    // 版本号（乐观锁）
    @Version
    private Integer version;

    /**
     * 创建应用版本
     */
    public static AppVersion create(CreateAppVersionCommand command) {
        // 1. 参数校验
        if (command.getPackageUrl() == null || command.getPackageUrl().isEmpty()) {
            throw new IllegalArgumentException("更新包URL不能为空");
        }
        if (command.getPackageHash() == null || command.getPackageHash().isEmpty()) {
            throw new IllegalArgumentException("更新包哈希值不能为空");
        }
        if (command.getPackageSize() == null || command.getPackageSize() <= 0) {
            throw new IllegalArgumentException("更新包大小必须大于0");
        }

        // 2. 创建版本实例
        AppVersion version = new AppVersion();
        version.versionId = IdGenerator.generateVersionId();
        version.appName = command.getAppName();
        version.version = command.getVersion();
        version.bundleVersion = command.getBundleVersion();
        version.platform = command.getPlatform();
        version.profileId = command.getProfileId();
        version.packageUrl = command.getPackageUrl();
        version.packageHash = command.getPackageHash();
        version.packageSize = command.getPackageSize();
        version.versionType = command.getVersionType();
        version.status = VersionStatus.DRAFT;
        version.changeLog = command.getChangeLog();
        version.forceUpdate = command.getForceUpdate() != null ? command.getForceUpdate() : false;
        version.minCompatibleVersion = command.getMinCompatibleVersion();
        version.dependency = command.getDependency();
        version.createdBy = command.getCreatedBy();
        version.createdAt = LocalDateTime.now();

        // 3. 发布领域事件
        version.addDomainEvent(new AppVersionCreatedEvent(version));
        return version;
    }

    /**
     * 发布版本
     */
    public void publish() {
        if (this.status != VersionStatus.DRAFT) {
            throw new IllegalStateException("只能发布草稿状态的版本");
        }

        // 校验版本信息完整性
        if (this.packageUrl == null || this.packageHash == null) {
            throw new IllegalStateException("版本信息不完整，无法发布");
        }

        this.status = VersionStatus.PUBLISHED;
        this.publishedAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new AppVersionPublishedEvent(this));
    }

    /**
     * 废弃版本
     */
    public void deprecate() {
        if (this.status == VersionStatus.DEPRECATED) {
            throw new IllegalStateException("版本已废弃");
        }

        this.status = VersionStatus.DEPRECATED;
        this.deprecatedAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new AppVersionDeprecatedEvent(this));
    }

    /**
     * 检查版本兼容性
     */
    public boolean isCompatibleWith(String currentVersion) {
        if (this.minCompatibleVersion == null) {
            return true;
        }
        return VersionComparator.compare(currentVersion, this.minCompatibleVersion) >= 0;
    }

    /**
     * 是否需要强制更新
     */
    public boolean requiresForceUpdate(String currentVersion) {
        if (this.forceUpdate) {
            return true;
        }
        // 如果当前版本低于最低兼容版本，也需要强制更新
        return !isCompatibleWith(currentVersion);
    }
}

/**
 * 版本类型
 */
public enum VersionType {
    FULL_PACKAGE,  // 完整包：包含所有应用代码和资源
    HOT_UPDATE     // 热更新包：仅包含增量更新内容
}

/**
 * 版本状态
 */
public enum VersionStatus {
    DRAFT,       // 草稿：版本创建但未发布
    PUBLISHED,   // 已发布：版本可用于升级发布
    DEPRECATED   // 已废弃：版本不再推荐使用
}

/**
 * 版本依赖
 */
public class VersionDependency {
    private String minNativeShellVersion;  // 最低原生壳版本
    private String minRnVersion;  // 最低 React Native 版本
    private List<String> requiredModules;  // 必需的原生模块

    public boolean isSatisfiedBy(TerminalInstance terminal) {
        // 检查终端是否满足依赖条件
        if (minNativeShellVersion != null) {
            String terminalShellVersion = terminal.getNativeShellVersion();
            if (VersionComparator.compare(terminalShellVersion, minNativeShellVersion) < 0) {
                return false;
            }
        }
        return true;
    }
}
```

```java
/**
 * 版本比较器
 */
public class VersionComparator {

    /**
     * 比较两个语义化版本号
     * @return 正数表示 v1 > v2，0表示相等，负数表示 v1 < v2
     */
    public static int compare(String v1, String v2) {
        if (v1 == null || v2 == null) {
            throw new IllegalArgumentException("版本号不能为空");
        }

        String[] parts1 = v1.split("\\.");
        String[] parts2 = v2.split("\\.");

        int maxLength = Math.max(parts1.length, parts2.length);
        for (int i = 0; i < maxLength; i++) {
            int num1 = i < parts1.length ? Integer.parseInt(parts1[i]) : 0;
            int num2 = i < parts2.length ? Integer.parseInt(parts2[i]) : 0;

            if (num1 != num2) {
                return num1 - num2;
            }
        }
        return 0;
    }
}
```

```java
@Entity
@Table(name = "upgrade_releases")
public class UpgradeRelease {

    @Id
    private String releaseId;  // 发布单唯一标识

    private String versionId;  // 关联的应用版本ID
    private String releaseName;  // 发布单名称，如"版本升级-2.3.1"

    // 目标终端
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private TargetScope targetScope;  // 目标终端范围（门店、设备模板等）

    // 灰度策略
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private GrayscaleStrategy grayscaleStrategy;  // 灰度发布策略

    @Enumerated(EnumType.STRING)
    private ReleaseStatus status;  // 发布状态

    private LocalDateTime scheduledAt;  // 计划发布时间
    private LocalDateTime startedAt;  // 实际开始时间
    private LocalDateTime completedAt;  // 完成时间

    // 统计信息
    private Integer totalCount;  // 总终端数
    private Integer successCount;  // 成功数
    private Integer failedCount;  // 失败数
    private Integer pendingCount;  // 待执行数

    // 回滚信息
    private String rollbackReason;  // 回滚原因
    private LocalDateTime rolledBackAt;  // 回滚时间

    private String createdBy;  // 创建人
    private LocalDateTime createdAt;  // 创建时间

    // 版本号（乐观锁）
    @Version
    private Integer version;

    /**
     * 创建升级发布
     */
    public static UpgradeRelease create(CreateUpgradeReleaseCommand command) {
        // 1. 参数校验
        if (command.getVersionId() == null || command.getVersionId().isEmpty()) {
            throw new IllegalArgumentException("版本ID不能为空");
        }
        if (command.getTargetScope() == null) {
            throw new IllegalArgumentException("目标范围不能为空");
        }

        // 2. 创建发布单
        UpgradeRelease release = new UpgradeRelease();
        release.releaseId = IdGenerator.generateReleaseId();
        release.versionId = command.getVersionId();
        release.releaseName = command.getReleaseName();
        release.targetScope = command.getTargetScope();
        release.grayscaleStrategy = command.getGrayscaleStrategy();
        release.status = ReleaseStatus.PENDING;
        release.scheduledAt = command.getScheduledAt();
        release.createdBy = command.getCreatedBy();
        release.createdAt = LocalDateTime.now();

        // 3. 发布领域事件
        release.addDomainEvent(new UpgradeReleaseCreatedEvent(release));
        return release;
    }

    /**
     * 开始发布
     */
    public void start() {
        if (this.status != ReleaseStatus.PENDING) {
            throw new IllegalStateException("只能启动待发布状态的发布单");
        }

        this.status = ReleaseStatus.IN_PROGRESS;
        this.startedAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new UpgradeReleaseStartedEvent(this));
    }

    /**
     * 完成发布
     */
    public void complete() {
        if (this.status != ReleaseStatus.IN_PROGRESS) {
            throw new IllegalStateException("只能完成发布中状态的发布单");
        }

        this.status = ReleaseStatus.COMPLETED;
        this.completedAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new UpgradeReleaseCompletedEvent(this));
    }

    /**
     * 发布失败
     */
    public void fail(String reason) {
        this.status = ReleaseStatus.FAILED;
        this.completedAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new UpgradeReleaseFailedEvent(this, reason));
    }

    /**
     * 回滚发布
     */
    public void rollback(String reason) {
        if (this.status != ReleaseStatus.IN_PROGRESS &&
            this.status != ReleaseStatus.COMPLETED) {
            throw new IllegalStateException("只能回滚发布中或已完成状态的发布单");
        }

        this.status = ReleaseStatus.ROLLED_BACK;
        this.rollbackReason = reason;
        this.rolledBackAt = LocalDateTime.now();

        // 发布领域事件
        this.addDomainEvent(new UpgradeReleaseRolledBackEvent(this, reason));
    }

    /**
     * 更新统计
     */
    public void updateStats(int total, int success, int failed) {
        this.totalCount = total;
        this.successCount = success;
        this.failedCount = failed;
        this.pendingCount = total - success - failed;

        // 检查是否全部完成
        if (this.pendingCount == 0 && this.status == ReleaseStatus.IN_PROGRESS) {
            if (this.failedCount == 0) {
                this.complete();
            } else if (this.successCount == 0) {
                this.fail("所有终端升级失败");
            } else {
                this.complete();
            }
        }
    }

    /**
     * 计算成功率
     */
    public double getSuccessRate() {
        if (totalCount == null || totalCount == 0) {
            return 0.0;
        }
        return (double) successCount / totalCount * 100;
    }

    /**
     * 是否可以回滚
     */
    public boolean canRollback() {
        return this.status == ReleaseStatus.IN_PROGRESS
            || this.status == ReleaseStatus.COMPLETED;
    }
}

/**
 * 发布状态
 */
public enum ReleaseStatus {
    PENDING,      // 待发布
    IN_PROGRESS,  // 发布中
    COMPLETED,    // 已完成
    FAILED,       // 失败
    ROLLED_BACK   // 已回滚
}
```

```java
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")

    private String errorCode;  // 错误码

    /**
     * 创建任务实例
     */
    public static UpgradeTaskInstance create(
            String releaseId,
            String terminalId,
            String versionId) {

        UpgradeTaskInstance instance = new UpgradeTaskInstance();
        instance.taskInstanceId = IdGenerator.generateTaskInstanceId();
        instance.releaseId = releaseId;
        instance.terminalId = terminalId;
        instance.versionId = versionId;
        instance.status = TaskInstanceStatus.PENDING;
        instance.createdAt = LocalDateTime.now();
        instance.downloadProgress = 0;
        instance.installProgress = 0;
        return instance;
    }

    /**
     * 开始下载
     */
    public void startDownload() {
        if (this.status != TaskInstanceStatus.PENDING) {
            throw new IllegalStateException("只能从待执行状态开始下载");
        }
        this.status = TaskInstanceStatus.DOWNLOADING;
        this.startedAt = LocalDateTime.now();
    }

    /**
     * 更新下载进度
     */
    public void updateDownloadProgress(int progress) {
        if (this.status != TaskInstanceStatus.DOWNLOADING) {
            throw new IllegalStateException("当前状态不是下载中");
        }
        if (progress < 0 || progress > 100) {
            throw new IllegalArgumentException("进度必须在0-100之间");
        }
        this.downloadProgress = progress;
    }

    /**
     * 开始安装
     */
    public void startInstall() {
        if (this.status != TaskInstanceStatus.DOWNLOADING) {
            throw new IllegalStateException("只能从下载中状态开始安装");
        }
        this.status = TaskInstanceStatus.INSTALLING;
    }

    /**
     * 更新安装进度
     */
    public void updateInstallProgress(int progress) {
        if (this.status != TaskInstanceStatus.INSTALLING) {
            throw new IllegalStateException("当前状态不是安装中");
        }
        if (progress < 0 || progress > 100) {
            throw new IllegalArgumentException("进度必须在0-100之间");
        }
        this.installProgress = progress;
    }

    /**
     * 标记成功
     */
    public void markSuccess(Map<String, Object> result) {
        this.status = TaskInstanceStatus.SUCCESS;
        this.completedAt = LocalDateTime.now();
        this.result = result;
        this.downloadProgress = 100;
        this.installProgress = 100;
    }

    /**
     * 标记失败
     */
    public void markFailed(String errorCode, String errorMessage) {
        this.status = TaskInstanceStatus.FAILED;
        this.completedAt = LocalDateTime.now();
        this.errorCode = errorCode;
        this.errorMessage = errorMessage;
    }
}

    DOWNLOADING,  // 下载中
    INSTALLING,   // 安装中

    FAILED        // 失败
```

## 5. 热更新应用服务

```java
@Service
@Transactional
public class VersionApplicationService {

    @Autowired
    private AppVersionRepository appVersionRepository;

    @Autowired
    private UpgradeReleaseRepository upgradeReleaseRepository;

    @Autowired
    private UpgradeTaskInstanceRepository upgradeTaskInstanceRepository;

    @Autowired
    private TerminalInstanceRepository terminalInstanceRepository;

    @Autowired
    private TdpClient tdpClient;

    @Autowired
    private ObjectStorageService objectStorageService;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 上传热更新包
     */
    public UploadPackageResponse uploadPackage(UploadPackageRequest request) {
        // 1. 校验文件格式
        if (!isValidPackageFormat(request.getFileName())) {
            throw new InvalidPackageFormatException("不支持的文件格式");
        }

        // 2. 校验文件大小（限制100MB）
        if (request.getPackageFile().length > 100 * 1024 * 1024) {
            throw new PackageTooLargeException("更新包大小不能超过100MB");
        }

        // 3. 上传到对象存储
        String packageUrl = objectStorageService.upload(
            request.getPackageFile(),
            "app-packages/" + request.getFileName()
        );

        // 4. 计算文件哈希
        String packageHash = HashUtils.sha256(request.getPackageFile());

        // 5. 记录上传日志
        log.info("上传热更新包成功: fileName={}, size={}, hash={}",
            request.getFileName(),
            request.getPackageFile().length,
            packageHash);

        return UploadPackageResponse.builder()
            .packageUrl(packageUrl)
            .packageHash(packageHash)
            .packageSize(request.getPackageFile().length)
            .build();
    }

    /**
     * 创建应用版本
     */
    public CreateAppVersionResponse createAppVersion(CreateAppVersionRequest request) {
        // 1. 校验版本号格式
        if (!isValidVersionFormat(request.getVersion())) {
            throw new InvalidVersionFormatException("版本号格式不正确");
        }

        // 2. 检查版本是否已存在
        Optional<AppVersion> existingVersion = appVersionRepository
            .findByAppNameAndVersionAndPlatform(
                request.getAppName(),
                request.getVersion(),
                request.getPlatform()
            );

        if (existingVersion.isPresent()) {
            throw new VersionAlreadyExistsException("版本已存在");
        }

        // 3. 创建应用版本
        AppVersion appVersion = AppVersion.create(CreateAppVersionCommand.builder()
            .appName(request.getAppName())
            .version(request.getVersion())
            .bundleVersion(request.getBundleVersion())
            .platform(request.getPlatform())
            .profileId(request.getProfileId())
            .packageUrl(request.getPackageUrl())
            .packageHash(request.getPackageHash())
            .packageSize(request.getPackageSize())
            .versionType(request.getVersionType())
            .changeLog(request.getChangeLog())
            .forceUpdate(request.getForceUpdate())
            .minCompatibleVersion(request.getMinCompatibleVersion())
            .dependency(request.getDependency())
            .createdBy(request.getCreatedBy())
            .build());

        appVersionRepository.save(appVersion);

        // 4. 发布领域事件
        eventPublisher.publish(appVersion.getDomainEvents());

        return CreateAppVersionResponse.builder()
            .versionId(appVersion.getVersionId())
            .version(appVersion.getVersion())
            .status(appVersion.getStatus())
            .createdAt(appVersion.getCreatedAt())
            .build();
    }

    /**
     * 发布应用版本
     */
    public void publishAppVersion(String versionId) {
        // 1. 查询版本
        AppVersion appVersion = appVersionRepository
            .findById(versionId)
            .orElseThrow(() -> new AppVersionNotFoundException());

        // 2. 发布版本
        appVersion.publish();
        appVersionRepository.save(appVersion);

        // 3. 发布领域事件
        eventPublisher.publish(appVersion.getDomainEvents());
    }

    /**
     * 创建升级发布
     */
    public CreateUpgradeReleaseResponse createUpgradeRelease(
            CreateUpgradeReleaseRequest request) {

        // 1. 查询应用版本
        AppVersion appVersion = appVersionRepository
            .findById(request.getVersionId())
            .orElseThrow(() -> new AppVersionNotFoundException());

        if (appVersion.getStatus() != VersionStatus.PUBLISHED) {
            throw new IllegalStateException("只能发布已发布状态的应用版本");
        }

        // 2. 创建发布单
        UpgradeRelease release = UpgradeRelease.create(
            CreateUpgradeReleaseCommand.builder()
                .versionId(appVersion.getVersionId())
                .releaseName(request.getReleaseName())
                .targetScope(request.getTargetScope())
                .grayscaleStrategy(request.getGrayscaleStrategy())
                .scheduledAt(request.getScheduledAt())
                .createdBy(request.getCreatedBy())
                .build()
        );

        upgradeReleaseRepository.save(release);

        // 3. 查询目标终端
        List<TerminalInstance> allTerminals = terminalInstanceRepository
            .findByTargetScope(request.getTargetScope());

        // 4. 应用灰度策略
        List<TerminalInstance> targetTerminals = request.getGrayscaleStrategy()
            .selectTargets(allTerminals);

        // 5. 创建任务实例
        for (TerminalInstance terminal : targetTerminals) {
            // 检查版本兼容性
            if (!appVersion.isCompatibleWith(terminal.getCurrentAppVersion())) {
                log.warn("终端版本不兼容，跳过: terminalId={}, currentVersion={}",
                    terminal.getTerminalId(),
                    terminal.getCurrentAppVersion());
                continue;
            }

            UpgradeTaskInstance taskInstance = UpgradeTaskInstance.create(
                release.getReleaseId(),
                terminal.getTerminalId(),
                appVersion.getVersionId()
            );
            upgradeTaskInstanceRepository.save(taskInstance);
        }

        release.updateStats(targetTerminals.size(), 0, 0);
        upgradeReleaseRepository.save(release);

        // 6. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());

        return CreateUpgradeReleaseResponse.builder()
            .releaseId(release.getReleaseId())
            .totalCount(targetTerminals.size())
            .build();
    }

    /**
     * 执行升级发布
     */
    public void executeUpgradeRelease(String releaseId) {
        // 1. 查询发布单
        UpgradeRelease release = upgradeReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new UpgradeReleaseNotFoundException());

        // 2. 开始发布
        release.start();
        upgradeReleaseRepository.save(release);

        // 3. 查询应用版本
        AppVersion appVersion = appVersionRepository
            .findById(release.getVersionId())
            .orElseThrow(() -> new AppVersionNotFoundException());

        // 4. 查询任务实例
        List<UpgradeTaskInstance> taskInstances = upgradeTaskInstanceRepository
            .findByReleaseId(releaseId);

        // 5. 委托 TDP 投递升级任务
        for (UpgradeTaskInstance taskInstance : taskInstances) {
            tdpClient.publishUpgradeTask(TdpUpgradeTask.builder()
                .taskId(taskInstance.getTaskInstanceId())
                .terminalId(taskInstance.getTerminalId())
                .versionId(appVersion.getVersionId())
                .version(appVersion.getVersion())
                .packageUrl(appVersion.getPackageUrl())
                .packageHash(appVersion.getPackageHash())
                .packageSize(appVersion.getPackageSize())
                .forceUpdate(appVersion.getForceUpdate())
                .priority(TaskPriority.NORMAL)
                .expiresAt(LocalDateTime.now().plusDays(7))
                .build());
        }

        // 6. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());
    }

    /**
     * 回报升级结果
     */
    public void reportUpgradeResult(ReportUpgradeResultRequest request) {
        // 1. 查询任务实例
        UpgradeTaskInstance taskInstance = upgradeTaskInstanceRepository
            .findById(request.getTaskInstanceId())
            .orElseThrow(() -> new UpgradeTaskInstanceNotFoundException());

        // 2. 更新任务实例
        if (request.getStatus() == TaskInstanceStatus.SUCCESS) {
            taskInstance.markSuccess(request.getResult());
        } else {
            taskInstance.markFailed(request.getErrorCode(), request.getErrorMessage());
        }
        upgradeTaskInstanceRepository.save(taskInstance);

        // 3. 更新发布单统计
        UpgradeRelease release = upgradeReleaseRepository
            .findById(taskInstance.getReleaseId())
            .orElseThrow(() -> new UpgradeReleaseNotFoundException());

        int successCount = upgradeTaskInstanceRepository
            .countByReleaseIdAndStatus(
                taskInstance.getReleaseId(),
                TaskInstanceStatus.SUCCESS
            );
        int failedCount = upgradeTaskInstanceRepository
            .countByReleaseIdAndStatus(
                taskInstance.getReleaseId(),
                TaskInstanceStatus.FAILED
            );

        release.updateStats(release.getTotalCount(), successCount, failedCount);
        upgradeReleaseRepository.save(release);

        // 4. 如果成功，更新终端版本信息
        if (request.getStatus() == TaskInstanceStatus.SUCCESS) {
            TerminalInstance terminal = terminalInstanceRepository
                .findById(taskInstance.getTerminalId())
                .orElseThrow(() -> new TerminalInstanceNotFoundException());

            AppVersion appVersion = appVersionRepository
                .findById(taskInstance.getVersionId())
                .orElseThrow(() -> new AppVersionNotFoundException());

            terminal.updateVersion(
                appVersion.getVersion(),
                appVersion.getBundleVersion(),
                terminal.getCurrentConfigVersion()
            );
            terminalInstanceRepository.save(terminal);
        }

        // 5. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());
    }

    /**
     * 校验文件格式
     */
    private boolean isValidPackageFormat(String fileName) {
        return fileName.endsWith(".zip") || fileName.endsWith(".apk");
    }

    /**
     * 校验版本号格式
     */
    private boolean isValidVersionFormat(String version) {
        return version.matches("\\d+\\.\\d+\\.\\d+");
    }
}
```

```java
public interface GrayscaleStrategy {

    /**
     * 从所有终端中选择目标终端

    TERMINAL_LIST, // 按终端列表：指定具体终端ID列表
```

### 6.2 按百分比灰度

```java
/**
 * 按百分比灰度策略
 * 从所有终端中随机选择指定百分比的终端进行发布
 */
public class PercentageGrayscaleStrategy implements GrayscaleStrategy {

    private int percentage;  // 百分比，取值范围 1-100

    public PercentageGrayscaleStrategy(int percentage) {
        this.percentage = percentage;
        validate();
    }

    @Override
    public List<TerminalInstance> selectTargets(
            List<TerminalInstance> allTerminals) {

        if (allTerminals == null || allTerminals.isEmpty()) {
            return Collections.emptyList();
        }

        // 计算目标数量，向上取整确保至少选择一个
        int targetCount = (int) Math.ceil(
            allTerminals.size() * percentage / 100.0
        );

        // 确保不超过总数
        targetCount = Math.min(targetCount, allTerminals.size());

        // 随机打乱顺序
        List<TerminalInstance> shuffled = new ArrayList<>(allTerminals);
        Collections.shuffle(shuffled);

        // 返回前 N 个
        return shuffled.subList(0, targetCount);
    }

    @Override
    public GrayscaleType getType() {
        return GrayscaleType.PERCENTAGE;
    }

    @Override
    public void validate() {
        if (percentage < 1 || percentage > 100) {
            throw new IllegalArgumentException(
                "百分比必须在 1-100 之间，当前值：" + percentage
            );
        }
    }
}
```

### 6.4 按批次灰度

```java
/**
 * 按批次灰度策略
 * 将终端分成多个批次，逐批发布
 */
public class BatchGrayscaleStrategy implements GrayscaleStrategy {

    private int batchSize;      // 每批次的终端数量
    private int currentBatch;   // 当前批次号（从 1 开始）
    private int totalBatches;   // 总批次数

    public BatchGrayscaleStrategy(int batchSize, int currentBatch) {
        this.batchSize = batchSize;
        this.currentBatch = currentBatch;
        validate();
    }

    @Override
    public List<TerminalInstance> selectTargets(
            List<TerminalInstance> allTerminals) {

        if (allTerminals == null || allTerminals.isEmpty()) {
            return Collections.emptyList();
        }

        // 计算总批次数
        this.totalBatches = (int) Math.ceil(
            (double) allTerminals.size() / batchSize
        );

        // 当前批次超出范围，返回空列表
        if (currentBatch > totalBatches) {
            return Collections.emptyList();
        }

        // 计算当前批次的起始和结束索引
        int startIndex = (currentBatch - 1) * batchSize;
        int endIndex = Math.min(startIndex + batchSize, allTerminals.size());

        // 返回当前批次的终端
        return new ArrayList<>(allTerminals.subList(startIndex, endIndex));
    }

    @Override
    public GrayscaleType getType() {
        return GrayscaleType.BATCH;
    }

    @Override
    public void validate() {
        if (batchSize < 1) {
            throw new IllegalArgumentException(
                "批次大小必须大于 0，当前值：" + batchSize
            );
        }
        if (currentBatch < 1) {
            throw new IllegalArgumentException(
                "批次号必须大于 0，当前值：" + currentBatch
            );
        }
    }

    public int getTotalBatches() {
        return totalBatches;
    }

    public int getCurrentBatch() {
        return currentBatch;
    }
}
```

### 6.5 按门店优先级灰度

```java
/**
 * 按门店优先级灰度策略
 * 根据门店优先级进行发布，优先级高的门店先发布
 */
public class StorePriorityGrayscaleStrategy implements GrayscaleStrategy {

    private int maxStoreCount;  // 最多选择的门店数量
    private StoreRepository storeRepository;

    public StorePriorityGrayscaleStrategy(
            int maxStoreCount,
            StoreRepository storeRepository) {
        this.maxStoreCount = maxStoreCount;
        this.storeRepository = storeRepository;
        validate();
    }

    @Override
    public List<TerminalInstance> selectTargets(
            List<TerminalInstance> allTerminals) {

        if (allTerminals == null || allTerminals.isEmpty()) {
            return Collections.emptyList();
        }

        // 按门店分组
        Map<String, List<TerminalInstance>> terminalsByStore = allTerminals
            .stream()
            .collect(Collectors.groupingBy(TerminalInstance::getStoreId));

        // 查询门店优先级
        List<String> storeIds = new ArrayList<>(terminalsByStore.keySet());
        Map<String, Integer> storePriorities = storeRepository
            .findPrioritiesByStoreIds(storeIds);

        // 按优先级排序门店（优先级数字越小越优先）
        List<String> sortedStoreIds = storeIds.stream()
            .sorted(Comparator.comparingInt(
                storeId -> storePriorities.getOrDefault(storeId, Integer.MAX_VALUE)
            ))
            .limit(maxStoreCount)
            .collect(Collectors.toList());

        // 收集选中门店的所有终端
        List<TerminalInstance> selectedTerminals = new ArrayList<>();
        for (String storeId : sortedStoreIds) {
            selectedTerminals.addAll(terminalsByStore.get(storeId));
        }

        return selectedTerminals;
    }

    @Override
    public GrayscaleType getType() {
        return GrayscaleType.STORE_PRIORITY;
    }

    @Override
    public void validate() {
        if (maxStoreCount < 1) {
            throw new IllegalArgumentException(
                "门店数量必须大于 0，当前值：" + maxStoreCount
            );
        }
        if (storeRepository == null) {
            throw new IllegalArgumentException("StoreRepository 不能为空");
        }
    }
}
```

### 6.7 灰度策略工厂

```java
/**
 * 灰度策略工厂
 * 根据配置创建对应的灰度策略实例

@Component
public class GrayscaleStrategyFactory {

    @Autowired

    /**

     */
    public GrayscaleStrategy createStrategy(GrayscaleStrategyConfig config) {
        if (config == null) {
            throw new IllegalArgumentException("灰度策略配置不能为空");
        }

        switch (config.getType()) {
            case ALL:
                return new AllGrayscaleStrategy();

            case PERCENTAGE:
                if (config.getPercentage() == null) {
                    throw new IllegalArgumentException("百分比灰度策略缺少 percentage 参数");
                }
                return new PercentageGrayscaleStrategy(config.getPercentage());

            case TERMINAL_LIST:
                if (config.getTerminalIds() == null || config.getTerminalIds().isEmpty()) {
                    throw new IllegalArgumentException("终端列表灰度策略缺少 terminalIds 参数");
                }
                return new TerminalListGrayscaleStrategy(config.getTerminalIds());

            case BATCH:
                if (config.getBatchSize() == null || config.getCurrentBatch() == null) {
                    throw new IllegalArgumentException("批次灰度策略缺少 batchSize 或 currentBatch 参数");
                }
                return new BatchGrayscaleStrategy(
                    config.getBatchSize(),
                    config.getCurrentBatch()
                );

            case STORE_PRIORITY:
                if (config.getMaxStoreCount() == null) {
                    throw new IllegalArgumentException("门店优先级灰度策略缺少 maxStoreCount 参数");
                }
                return new StorePriorityGrayscaleStrategy(
                    config.getMaxStoreCount(),
                    storeRepository
                );

            default:
                throw new IllegalArgumentException(
                    "不支持的灰度策略类型：" + config.getType()
                );
        }
    }
}

/**
 * 灰度策略配置
 */
@Data
public class GrayscaleStrategyConfig {
    private GrayscaleType type;
    private Integer percentage;
    private List<String> terminalIds;
    private Integer batchSize;
    private Integer currentBatch;
    private Integer maxStoreCount;
}
```

## 7. 版本回滚机制

### 7.1 回滚服务

```java
@Service
@Transactional
public class VersionRollbackService {

    @Autowired
    private UpgradeReleaseRepository upgradeReleaseRepository;

    @Autowired
    private AppVersionRepository appVersionRepository;

    @Autowired
    private TerminalInstanceRepository terminalInstanceRepository;

    @Autowired
    private UpgradeTaskInstanceRepository upgradeTaskInstanceRepository;

    @Autowired
    private TdpClient tdpClient;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 回滚到上一个稳定版本
     */
    public RollbackResult rollbackToLastStable(String releaseId, String reason) {
        // 1. 查询发布单
        UpgradeRelease release = upgradeReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new UpgradeReleaseNotFoundException());

        // 2. 校验发布单状态
        if (!release.canRollback()) {
            throw new IllegalStateException(
                "发布单状态不允许回滚：" + release.getStatus()
            );
        }

        // 3. 查询当前版本
        AppVersion currentVersion = appVersionRepository
            .findById(release.getVersionId())
            .orElseThrow(() -> new AppVersionNotFoundException());

        // 4. 查询上一个稳定版本
        AppVersion lastStableVersion = appVersionRepository
            .findLastStableVersion(
                currentVersion.getAppName(),
                currentVersion.getPlatform(),
                currentVersion.getProfileId()
            );

        if (lastStableVersion == null) {
            throw new NoStableVersionException("没有可回滚的稳定版本");
        }

        // 5. 查询需要回滚的终端（已成功升级的终端）
        List<UpgradeTaskInstance> successTasks = upgradeTaskInstanceRepository
            .findByReleaseIdAndStatus(releaseId, TaskInstanceStatus.SUCCESS);

        List<String> terminalIds = successTasks.stream()
            .map(UpgradeTaskInstance::getTerminalId)
            .collect(Collectors.toList());

        List<TerminalInstance> terminals = terminalInstanceRepository
            .findByTerminalIdIn(terminalIds);

        // 6. 创建回滚任务
        int rollbackCount = 0;
        for (TerminalInstance terminal : terminals) {
            try {
                // 委托 TDP 投递回滚任务
                tdpClient.publishUpgradeTask(TdpUpgradeTask.builder()
                    .taskId(IdGenerator.generateTaskId())
                    .terminalId(terminal.getTerminalId())
                    .versionId(lastStableVersion.getVersionId())
                    .version(lastStableVersion.getVersion())
                    .packageUrl(lastStableVersion.getPackageUrl())
                    .packageHash(lastStableVersion.getPackageHash())
                    .packageSize(lastStableVersion.getPackageSize())
                    .forceUpdate(true)  // 回滚必须强制更新
                    .priority(TaskPriority.URGENT)  // 回滚任务优先级最高
                    .expiresAt(LocalDateTime.now().plusHours(1))
                    .build());

                rollbackCount++;
            } catch (Exception e) {
                log.error("投递回滚任务失败：terminalId={}",
                    terminal.getTerminalId(), e);
            }
        }

        // 7. 标记发布单为已回滚
        release.rollback(reason);
        upgradeReleaseRepository.save(release);

        // 8. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());

        // 9. 记录回滚日志
        log.info("版本回滚完成：releaseId={}, currentVersion={}, rollbackVersion={}, rollbackCount={}",
            releaseId,
            currentVersion.getVersion(),
            lastStableVersion.getVersion(),
            rollbackCount);

        return RollbackResult.builder()
            .releaseId(releaseId)
            .currentVersion(currentVersion.getVersion())
            .rollbackVersion(lastStableVersion.getVersion())
            .rollbackCount(rollbackCount)
            .reason(reason)
            .rolledBackAt(LocalDateTime.now())
            .build();
    }

    /**
     * 回滚到指定版本
     */
    public RollbackResult rollbackToVersion(
            String releaseId,
            String targetVersionId,
            String reason) {

        // 1. 查询发布单
        UpgradeRelease release = upgradeReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new UpgradeReleaseNotFoundException());

        // 2. 校验发布单状态
        if (!release.canRollback()) {
            throw new IllegalStateException(
                "发布单状态不允许回滚：" + release.getStatus()
            );
        }

        // 3. 查询目标版本
        AppVersion targetVersion = appVersionRepository
            .findById(targetVersionId)
            .orElseThrow(() -> new AppVersionNotFoundException());

        // 4. 校验目标版本状态
        if (targetVersion.getStatus() != VersionStatus.PUBLISHED) {
            throw new IllegalStateException(
                "目标版本未发布，无法回滚：" + targetVersion.getVersion()
            );
        }

        // 5. 查询需要回滚的终端
        List<UpgradeTaskInstance> successTasks = upgradeTaskInstanceRepository
            .findByReleaseIdAndStatus(releaseId, TaskInstanceStatus.SUCCESS);

        List<String> terminalIds = successTasks.stream()
            .map(UpgradeTaskInstance::getTerminalId)
            .collect(Collectors.toList());

        List<TerminalInstance> terminals = terminalInstanceRepository
            .findByTerminalIdIn(terminalIds);

        // 6. 创建回滚任务
        int rollbackCount = 0;
        for (TerminalInstance terminal : terminals) {
            try {
                tdpClient.publishUpgradeTask(TdpUpgradeTask.builder()
                    .taskId(IdGenerator.generateTaskId())
                    .terminalId(terminal.getTerminalId())
                    .versionId(targetVersion.getVersionId())
                    .version(targetVersion.getVersion())
                    .packageUrl(targetVersion.getPackageUrl())
                    .packageHash(targetVersion.getPackageHash())
                    .packageSize(targetVersion.getPackageSize())
                    .forceUpdate(true)
                    .priority(TaskPriority.URGENT)
                    .expiresAt(LocalDateTime.now().plusHours(1))
                    .build());

                rollbackCount++;
            } catch (Exception e) {
                log.error("投递回滚任务失败：terminalId={}",
                    terminal.getTerminalId(), e);
            }
        }

        // 7. 标记发布单为已回滚
        release.rollback(reason);
        upgradeReleaseRepository.save(release);

        // 8. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());

        return RollbackResult.builder()
            .releaseId(releaseId)
            .rollbackVersion(targetVersion.getVersion())
            .rollbackCount(rollbackCount)
            .reason(reason)
            .rolledBackAt(LocalDateTime.now())
            .build();
    }

    /**
     * 查询可回滚的版本列表
     */
    public List<AppVersion> findRollbackableVersions(String releaseId) {
        // 1. 查询发布单
        UpgradeRelease release = upgradeReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new UpgradeReleaseNotFoundException());

        // 2. 查询当前版本
        AppVersion currentVersion = appVersionRepository
            .findById(release.getVersionId())
            .orElseThrow(() -> new AppVersionNotFoundException());

        // 3. 查询所有已发布的历史版本（版本号小于当前版本）
        return appVersionRepository.findRollbackableVersions(
            currentVersion.getAppName(),
            currentVersion.getPlatform(),
            currentVersion.getProfileId(),
            currentVersion.getVersion()
        );
    }
}

/**
 * 回滚结果
 */
@Data
@Builder
public class RollbackResult {
    private String releaseId;
    private String currentVersion;
    private String rollbackVersion;
    private Integer rollbackCount;
    private String reason;
    private LocalDateTime rolledBackAt;
}
```

```java
public interface AppVersionRepository extends JpaRepository<AppVersion, String> {

     * 查询上一个稳定版本

        SELECT v FROM AppVersion v
        WHERE v.appName = :appName
```

## 8. 本文结论

热更新服务设计展示了完整的版本管理实现：

- **AppVersion**：应用版本聚合根，支持完整包和热更新包
- **UpgradeRelease**：升级发布聚合根，支持灰度发布、回滚
- **VersionApplicationService**：版本管理应用服务，支持上传、发布、回滚
- **灰度策略**：支持按百分比、按终端列表等多种灰度方式
- **版本回滚**：支持回滚到上一个稳定版本
