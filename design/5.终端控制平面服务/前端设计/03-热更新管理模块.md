# 热更新管理模块前端设计

## 一、模块概述

热更新管理模块负责终端应用版本的管理、更新任务的创建和执行、灰度发布策略的配置以及版本回滚功能。

## 二、页面结构

```
热更新管理
├── 版本管理 (/updates/versions)
├── 更新任务 (/updates/tasks)
├── 灰度发布 (/updates/grayscale)
└── 回滚管理 (/updates/rollback)
```

// src/pages/Update/Versions/index.tsx
import React from 'react'
import { Table, Card, Button, Space, Tag, Upload, Modal, Form, Input } from 'antd'
import { PlusOutlined, UploadOutlined } from '@ant-design/icons'

interface Version {
  id: string
  versionNumber: string
  packageType: 'FULL' | 'INCREMENTAL'
  fileSize: number
  status: 'DRAFT' | 'PUBLISHED' | 'DEPRECATED'
  createdAt: string
}

const VersionManagement: React.FC = () => {
  const [uploadModalVisible, setUploadModalVisible] = useState(false)
  const [form] = Form.useForm()

  const columns = [
    { title: '版本号', dataIndex: 'versionNumber', key: 'versionNumber' },
    {
      title: '包类型',
      dataIndex: 'packageType',
      key: 'packageType',
      render: (type: string) => (type === 'FULL' ? '全量包' : '增量包'),
    },
    {
      title: '文件大小',
      dataIndex: 'fileSize',
      key: 'fileSize',
      render: (size: number) => `${(size / 1024 / 1024).toFixed(2)} MB`,
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      render: (status: string) => {
        const statusConfig = {
          DRAFT: { color: 'default', text: '草稿' },
          PUBLISHED: { color: 'success', text: '已发布' },
          DEPRECATED: { color: 'warning', text: '已废弃' },
        }
        const config = statusConfig[status]
        return <Tag color={config.color}>{config.text}</Tag>
      },
    },
    { title: '创建时间', dataIndex: 'createdAt', key: 'createdAt' },
    {
      title: '操作',
      key: 'action',
      render: (_, record) => (
        <Space>
          <Button type="link" size="small" onClick={() => handlePublish(record)}>
            发布
          </Button>
          <Button type="link" size="small" onClick={() => handleCreateTask(record)}>
            创建任务
          </Button>
          <Button type="link" size="small" danger onClick={() => handleDeprecate(record)}>
            废弃
          </Button>
        </Space>
      ),
    },
  ]

  return (
    <Card>
      <div style={{ marginBottom: 16 }}>
        <Button type="primary" icon={<PlusOutlined />} onClick={() => setUploadModalVisible(true)}>
          上传版本
        </Button>
      </div>

      <Table columns={columns} rowKey="id" />

      {/* 上传版本弹窗 */}
      <Modal
        title="上传版本"
        open={uploadModalVisible}
        onOk={handleUpload}
        onCancel={() => setUploadModalVisible(false)}
      >
        <Form form={form} layout="vertical">
          <Form.Item name="versionNumber" label="版本号" rules={[{ required: true }]}>
            <Input placeholder="例如: 1.0.0" />
          </Form.Item>
          <Form.Item name="packageType" label="包类型" rules={[{ required: true }]}>
            <Select>
              <Select.Option value="FULL">全量包</Select.Option>
              <Select.Option value="INCREMENTAL">增量包</Select.Option>
            </Select>
          </Form.Item>
          <Form.Item name="file" label="安装包" rules={[{ required: true }]}>
            <Upload>
              <Button icon={<UploadOutlined />}>选择文件</Button>
            </Upload>
          </Form.Item>
          <Form.Item name="releaseNotes" label="更新说明">
            <Input.TextArea rows={4} placeholder="请输入更新说明" />
          </Form.Item>
        </Form>
      </Modal>
    </Card>
  )
}

## 七、API集成

```typescript
// src/api/update.ts
import request from '@/utils/request'

// 查询版本列表
export const queryVersions = (params: any) => {
  return request.get('/api/v1/updates/versions', { params })
}

// 上传版本
export const uploadVersion = (data: FormData) => {
  return request.post('/api/v1/updates/versions', data, {
    headers: { 'Content-Type': 'multipart/form-data' },
  })
}

// 发布版本
export const publishVersion = (id: string) => {
  return request.post(`/api/v1/updates/versions/${id}/publish`)
}

// 创建更新任务
export const createUpdateTask = (data: any) => {
  return request.post('/api/v1/updates/tasks', data)
}

// 查询更新任务列表
export const queryUpdateTasks = (params: any) => {
  return request.get('/api/v1/updates/tasks', { params })
}

// 开始任务
export const startUpdateTask = (id: string) => {
  return request.post(`/api/v1/updates/tasks/${id}/start`)
}

// 暂停任务
export const pauseUpdateTask = (id: string) => {
  return request.post(`/api/v1/updates/tasks/${id}/pause`)
}

// 配置灰度策略
export const configureGrayscale = (data: any) => {
  return request.post('/api/v1/updates/grayscale', data)
}

// 回滚版本
export const rollbackVersion = (data: { taskId: string; targetVersion: string; reason: string }) => {
  return request.post('/api/v1/updates/rollback', data)
}
```
