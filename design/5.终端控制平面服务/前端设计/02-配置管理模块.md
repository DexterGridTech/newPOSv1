# 配置管理模块前端设计

## 一、模块概述

配置管理模块负责终端配置的创建、下发、版本管理和审计，支持按组织、门店、终端维度进行配置管理。

## 二、页面结构

```
配置管理
├── 配置模板 (/configs/templates)
├── 配置下发 (/configs/distribution)
├── 配置历史 (/configs/history)
└── 配置审计 (/configs/audit)
```

## 四、配置下发页面

### 4.1 页面布局

```typescript
// src/pages/Config/Distribution/index.tsx
import React, { useState } from 'react'
import { Card, Steps, Form, Select, Button, Table, Result } from 'antd'
import { distributeConfig } from '@/api/config'

const ConfigDistribution: React.FC = () => {
  const [currentStep, setCurrentStep] = useState(0)
  const [form] = Form.useForm()
  const [selectedTargets, setSelectedTargets] = useState<string[]>([])

  const steps = [
    { title: '选择配置' },
    { title: '选择目标' },
    { title: '确认下发' },
    { title: '下发完成' },
  ]

  const handleNext = async () => {
    if (currentStep === 2) {
      // 执行下发
      const values = form.getFieldsValue()
      await distributeConfig({
        templateId: values.templateId,
        targetIds: selectedTargets,
        effectiveAt: values.effectiveAt,
      })
    }
    setCurrentStep(currentStep + 1)
  }

  return (
    <Card>
      <Steps current={currentStep} items={steps} style={{ marginBottom: 24 }} />

      {currentStep === 0 && (
        <Form form={form} layout="vertical">
          <Form.Item name="templateId" label="配置模板" rules={[{ required: true }]}>
            <Select placeholder="请选择配置模板">
              {/* 模板选项 */}
            </Select>
          </Form.Item>
          <Form.Item name="effectiveAt" label="生效时间">
            <Select placeholder="请选择">
              <Select.Option value="IMMEDIATE">立即生效</Select.Option>
              <Select.Option value="NEXT_RESTART">下次重启</Select.Option>
              <Select.Option value="SCHEDULED">定时生效</Select.Option>
            </Select>
          </Form.Item>
          <Button type="primary" onClick={handleNext}>
            下一步
          </Button>
        </Form>
      )}

      {currentStep === 1 && (
        <div>
          <Table
            rowSelection={{
              selectedRowKeys: selectedTargets,
              onChange: (keys) => setSelectedTargets(keys as string[]),
            }}
            columns={[
              { title: '终端ID', dataIndex: 'terminalId' },
              { title: '门店', dataIndex: 'storeName' },
              { title: '状态', dataIndex: 'status' },
            ]}
          />
          <div style={{ marginTop: 16 }}>
            <Button onClick={() => setCurrentStep(0)}>上一步</Button>
            <Button type="primary" onClick={handleNext} style={{ marginLeft: 8 }}>
              下一步
            </Button>
          </div>
        </div>
      )}

      {currentStep === 2 && (
        <div>
          <p>确认下发配置到 {selectedTargets.length} 个终端？</p>
          <Button onClick={() => setCurrentStep(1)}>上一步</Button>
          <Button type="primary" onClick={handleNext} style={{ marginLeft: 8 }}>
            确认下发
          </Button>
        </div>
      )}

      {currentStep === 3 && (
        <Result
          status="success"
          title="配置下发成功"
          subTitle={`已成功下发到 ${selectedTargets.length} 个终端`}
          extra={[
            <Button type="primary" key="history" onClick={() => window.location.href = '/configs/history'}>
              查看历史
            </Button>,
          ]}
        />
      )}
    </Card>
  )
}
```

## 七、API集成

```typescript
// src/api/config.ts
import request from '@/utils/request'

// 查询配置模板列表
export const queryConfigTemplates = (params: any) => {
  return request.get('/api/v1/configs/templates', { params })
}

// 创建配置模板
export const createConfigTemplate = (data: any) => {
  return request.post('/api/v1/configs/templates', data)
}

// 更新配置模板
export const updateConfigTemplate = (id: string, data: any) => {
  return request.put(`/api/v1/configs/templates/${id}`, data)
}

// 发布配置模板
export const publishConfigTemplate = (id: string) => {
  return request.post(`/api/v1/configs/templates/${id}/publish`)
}

// 下发配置
export const distributeConfig = (data: {
  templateId: string
  targetIds: string[]
  effectiveAt: string
}) => {
  return request.post('/api/v1/configs/distribute', data)
}

// 查询配置历史
export const queryConfigHistory = (params: any) => {
  return request.get('/api/v1/configs/history', { params })
}

// 查询配置审计日志
export const queryConfigAuditLogs = (params: any) => {
  return request.get('/api/v1/configs/audit-logs', { params })
}

// 查询配置差异
export const getConfigDiff = (id: string) => {
  return request.get(`/api/v1/configs/audit-logs/${id}/diff`)
}
```

## 八、状态管理

```typescript
// src/store/slices/configSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface ConfigState {
  currentTemplate: any | null
  templateList: any[]
  distributionHistory: any[]
  loading: boolean
}

const initialState: ConfigState = {
  currentTemplate: null,
  templateList: [],
  distributionHistory: [],
  loading: false,
}

const configSlice = createSlice({
  name: 'config',
  initialState,
  reducers: {
    setCurrentTemplate: (state, action: PayloadAction<any>) => {
      state.currentTemplate = action.payload
    },
    setTemplateList: (state, action: PayloadAction<any[]>) => {
      state.templateList = action.payload
    },
    setDistributionHistory: (state, action: PayloadAction<any[]>) => {
      state.distributionHistory = action.payload
    },
    setLoading: (state, action: PayloadAction<boolean>) => {
      state.loading = action.payload
    },
  },
})

export const { setCurrentTemplate, setTemplateList, setDistributionHistory, setLoading } = configSlice.actions
export default configSlice.reducer
```
