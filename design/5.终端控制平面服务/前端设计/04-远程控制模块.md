# 远程控制模块前端设计

## 一、模块概述

远程控制模块提供对终端设备的远程操作能力，包括重启、日志上传、强制同步、屏幕锁定、清除缓存、诊断等命令的执行和监控。

## 二、页面结构

```
远程控制
├── 控制任务 (/control/tasks)
├── 任务执行 (/control/execution)
├── 执行记录 (/control/records)
└── 审计日志 (/control/audit)
```

```typescript
import React from 'react'
import { Table, Card, Tag, Button, Space, Progress } from 'antd'

const ControlTaskList: React.FC = () => {
  const columns = [
    { title: '任务名称', dataIndex: 'taskName', key: 'taskName' },
    {
      title: '控制命令',
      dataIndex: 'command',
      key: 'command',
      render: (command: string) => {
        const commandMap = {
          RESTART: '重启设备',
          LOG_UPLOAD: '上传日志',
          FORCE_SYNC: '强制同步',
          SCREEN_LOCK: '锁定屏幕',
          CLEAR_CACHE: '清除缓存',
          DIAGNOSTICS: '系统诊断',
        }
        return commandMap[command] || command
      },
    },
    { title: '目标数量', dataIndex: 'targetCount', key: 'targetCount' },
    {
      title: '执行进度',
      key: 'progress',
      render: (_, record) => {
        const percent = (record.successCount / record.targetCount) * 100
        return (
          <div>
            <Progress percent={percent} size="small" />
            <div style={{ fontSize: 12, color: '#666' }}>
              成功: {record.successCount} / 失败: {record.failedCount} / 超时: {record.timeoutCount}
            </div>
          </div>
        )
      },
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      render: (status: string) => {
        const statusConfig = {
          PENDING: { color: 'default', text: '待执行' },
          IN_PROGRESS: { color: 'processing', text: '执行中' },
          SUCCESS: { color: 'success', text: '成功' },
          FAILED: { color: 'error', text: '失败' },
          TIMEOUT: { color: 'warning', text: '超时' },
        }
        const config = statusConfig[status]
        return <Tag color={config.color}>{config.text}</Tag>
      },
    },
    { title: '创建时间', dataIndex: 'createdAt', key: 'createdAt' },
    {
      title: '操作',
      key: 'action',
      render: (_, record) => (
        <Space>
          <Button type="link" size="small" onClick={() => handleExecute(record)}>
            执行
          </Button>
          <Button type="link" size="small" onClick={() => handleViewRecords(record)}>
            查看记录
          </Button>
          {record.failedCount > 0 && (
            <Button type="link" size="small" onClick={() => handleRetry(record)}>
              重试失败
            </Button>
          )}
          <Button type="link" size="small" danger onClick={() => handleCancel(record)}>
            取消
          </Button>
        </Space>
      ),
    },
  ]

  return (
    <Card>
      <div style={{ marginBottom: 16 }}>
        <Button type="primary" onClick={() => window.location.href = '/control/tasks/create'}>
          创建任务
        </Button>
      </div>
      <Table columns={columns} rowKey="id" />
    </Card>
  )
}
```

## 六、实时监控组件

### 6.1 任务执行监控

```typescript
// src/components/ControlMonitor/index.tsx
import React, { useEffect, useState } from 'react'
import { Card, Statistic, Row, Col, Progress } from 'antd'
import { CheckCircleOutlined, CloseCircleOutlined, ClockCircleOutlined } from '@ant-design/icons'

interface ControlMonitorProps {
  taskId: string
}

const ControlMonitor: React.FC<ControlMonitorProps> = ({ taskId }) => {
  const [stats, setStats] = useState({
    total: 0,
    success: 0,
    failed: 0,
    timeout: 0,
    pending: 0,
  })

  // WebSocket实时更新
  useEffect(() => {
    const ws = new WebSocket(`${WS_URL}/control/tasks/${taskId}/monitor`)

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data)
      setStats(data)
    }

    return () => ws.close()
  }, [taskId])

  const successRate = stats.total > 0 ? (stats.success / stats.total) * 100 : 0

  return (
    <Card title="执行监控">
      <Row gutter={16}>
        <Col span={6}>
          <Statistic title="总数" value={stats.total} />
        </Col>
        <Col span={6}>
          <Statistic
            title="成功"
            value={stats.success}
            valueStyle={{ color: '#3f8600' }}
            prefix={<CheckCircleOutlined />}
          />
        </Col>
        <Col span={6}>
          <Statistic
            title="失败"
            value={stats.failed}
            valueStyle={{ color: '#cf1322' }}
            prefix={<CloseCircleOutlined />}
          />
        </Col>
        <Col span={6}>
          <Statistic
            title="超时"
            value={stats.timeout}
            valueStyle={{ color: '#faad14' }}
            prefix={<ClockCircleOutlined />}
          />
        </Col>
      </Row>

      <div style={{ marginTop: 24 }}>
        <div>成功率</div>
        <Progress percent={successRate} status={successRate === 100 ? 'success' : 'active'} />
      </div>
    </Card>
  )
}

export default ControlMonitor
```

## 七、API集成

```typescript
// src/api/control.ts
import request from '@/utils/request'

// 创建控制任务
export const createControlTask = (data: any) => {
  return request.post('/api/v1/control-tasks', data)
}

// 执行任务
export const executeControlTask = (taskId: string) => {
  return request.post(`/api/v1/control-tasks/${taskId}/execute`)
}

// 查询任务列表
export const queryControlTasks = (params: any) => {
  return request.get('/api/v1/control-tasks', { params })
}

// 查询任务详情
export const getControlTaskDetail = (taskId: string) => {
  return request.get(`/api/v1/control-tasks/${taskId}`)
}

// 查询执行记录
export const queryExecutionRecords = (params: any) => {
  return request.get('/api/v1/control-tasks/records', { params })
}

// 上报执行结果
export const reportExecutionResult = (data: any) => {
  return request.post('/api/v1/control-tasks/results', data)
}

// 取消任务
export const cancelControlTask = (taskId: string) => {
  return request.post(`/api/v1/control-tasks/${taskId}/cancel`)
}

// 重试失败记录
export const retryFailedRecords = (taskId: string) => {
  return request.post(`/api/v1/control-tasks/${taskId}/retry`)
}

// 查询审计日志
export const queryAuditLogs = (params: any) => {
  return request.get('/api/v1/control-tasks/audit-logs', { params })
}
```
