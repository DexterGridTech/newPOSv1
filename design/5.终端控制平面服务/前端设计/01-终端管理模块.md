# 终端管理模块前端设计

## 一、模块概述

终端管理模块是TCP管理后台的核心模块，负责终端设备的全生命周期管理，包括终端列表查询、详情查看、激活管理、绑定管理等功能。

## 二、页面结构

### 2.1 页面导航

```
终端管理
├── 终端列表 (/terminals)
├── 终端详情 (/terminals/:id)
├── 激活管理 (/terminals/activation)
└── 绑定管理 (/terminals/binding)
```

// src/pages/Terminal/Detail/index.tsx
import React, { useEffect, useState } from 'react'
import { useParams } from 'react-router-dom'
import { Card, Descriptions, Tabs, Tag, Button, Space } from 'antd'
import { getTerminalDetail } from '@/api/terminal'

const TerminalDetail: React.FC = () => {
  const { id } = useParams<{ id: string }>()

  const [loading, setLoading] = useState(false)

  useEffect(() => {
    loadTerminalDetail()
  }, [id])

    setLoading(true)

      setLoading(false)

## 五、激活管理页面

### 5.1 激活流程

```typescript
// src/pages/Terminal/Activation/index.tsx
import React, { useState } from 'react'
import { Card, Form, Input, Button, Steps, Result } from 'antd'
import { activateTerminal } from '@/api/terminal'

const TerminalActivation: React.FC = () => {
  const [form] = Form.useForm()
  const [currentStep, setCurrentStep] = useState(0)
  const [activationCode, setActivationCode] = useState('')

  const handleGenerateCode = async () => {
    const values = await form.validateFields(['terminalId', 'storeId'])
    // 调用API生成激活码
    const code = await generateActivationCode(values)
    setActivationCode(code)
    setCurrentStep(1)
  }

  const handleActivate = async () => {
    // 调用激活API
    await activateTerminal({ activationCode })
    setCurrentStep(2)
  }

  return (
    <Card>
      <Steps
        current={currentStep}
        items={[
          { title: '填写信息' },
          { title: '生成激活码' },
          { title: '激活完成' },
        ]}
        style={{ marginBottom: 24 }}
      />

      {currentStep === 0 && (
        <Form form={form} layout="vertical">
          <Form.Item name="terminalId" label="终端ID" rules={[{ required: true }]}>
            <Input placeholder="请输入终端ID" />
          </Form.Item>
          <Form.Item name="storeId" label="门店ID" rules={[{ required: true }]}>
            <Input placeholder="请输入门店ID" />
          </Form.Item>
          <Button type="primary" onClick={handleGenerateCode}>
            生成激活码
          </Button>
        </Form>
      )}

      {currentStep === 1 && (
        <div>
          <p>激活码已生成，请在终端设备上输入以下激活码：</p>
          <div style={{ fontSize: 24, fontWeight: 'bold', margin: '20px 0' }}>
            {activationCode}
          </div>
          <Button type="primary" onClick={handleActivate}>
            确认激活
          </Button>
        </div>
      )}

      {currentStep === 2 && (
        <Result
          status="success"
          title="激活成功"
          subTitle="终端已成功激活，可以正常使用"
          extra={[
            <Button type="primary" key="list" onClick={() => window.location.href = '/terminals'}>
              返回列表
            </Button>,
          ]}
        />
      )}
    </Card>
  )
}
```

## 七、状态管理

```typescript
// src/store/slices/terminalSlice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface TerminalState {
  currentTerminal: any | null
  terminalList: any[]
  loading: boolean
}

const initialState: TerminalState = {
  currentTerminal: null,
  terminalList: [],
  loading: false,
}

const terminalSlice = createSlice({
  name: 'terminal',
  initialState,
  reducers: {
    setCurrentTerminal: (state, action: PayloadAction<any>) => {
      state.currentTerminal = action.payload
    },
    setTerminalList: (state, action: PayloadAction<any[]>) => {
      state.terminalList = action.payload
    },
    setLoading: (state, action: PayloadAction<boolean>) => {
      state.loading = action.payload
    },
  },
})

export const { setCurrentTerminal, setTerminalList, setLoading } = terminalSlice.actions
export default terminalSlice.reducer
```
