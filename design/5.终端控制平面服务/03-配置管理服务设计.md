# 03-配置管理服务设计

## 1. 配置管理概述

配置管理是 TCP 的核心功能之一，负责：
- 配置模型定义与版本管理
- 配置发布与灰度策略
- 配置回滚与审计追踪
- 终端配置同步与应用

**核心职责**：
- 管理终端配置的全生命周期
- 支持配置的版本化管理
- 提供灰度发布能力
- 支持配置回滚
- 记录配置变更审计日志

### 2.1 领域模型

```java
@Entity
@Table(name = "config_versions")
public class ConfigVersion {

    @Id
    private String configVersionId;

    private String configName;
    private String version;  // cfg-20260406-001

    // 配置内容
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private Map<String, Object> configData;

    // 配置 Schema
    private String schemaId;

    // 目标范围
    private String projectId;
    private String profileId;

    @Enumerated(EnumType.STRING)
    private ConfigStatus status;  // DRAFT, PUBLISHED, DEPRECATED

    // 变更说明
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private List<String> changeLog;

    // 基线版本（用于回滚）
    private String baselineVersionId;

    private String createdBy;
    private LocalDateTime createdAt;
    private LocalDateTime publishedAt;
    private LocalDateTime deprecatedAt;

    @Version
    private Integer version;

    // 领域方法

    /**
     * 创建配置版本
     */
    public static ConfigVersion create(CreateConfigVersionCommand command) {
        ConfigVersion config = new ConfigVersion();
        config.configVersionId = IdGenerator.generateConfigVersionId();
        config.configName = command.getConfigName();
        config.version = generateVersionNumber();
        config.configData = command.getConfigData();
        config.schemaId = command.getSchemaId();
        config.projectId = command.getProjectId();
        config.profileId = command.getProfileId();
        config.status = ConfigStatus.DRAFT;
        config.changeLog = command.getChangeLog();
        config.baselineVersionId = command.getBaselineVersionId();
        config.createdBy = command.getCreatedBy();
        config.createdAt = LocalDateTime.now();

        config.addDomainEvent(new ConfigVersionCreatedEvent(config));
        return config;
    }

    /**
     * 发布配置
     */
    public void publish() {
        if (this.status != ConfigStatus.DRAFT) {
            throw new IllegalStateException("只能发布草稿状态的配置");
        }
        this.status = ConfigStatus.PUBLISHED;
        this.publishedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigVersionPublishedEvent(this));
    }

    /**
     * 废弃配置
     */
    public void deprecate() {
        if (this.status == ConfigStatus.DEPRECATED) {
            throw new IllegalStateException("配置已废弃");
        }
        this.status = ConfigStatus.DEPRECATED;
        this.deprecatedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigVersionDeprecatedEvent(this));
    }

    /**
     * 更新配置数据
     */
    public void updateConfigData(Map<String, Object> newConfigData, String updatedBy) {
        if (this.status != ConfigStatus.DRAFT) {
            throw new IllegalStateException("只能修改草稿状态的配置");
        }
        this.configData = newConfigData;
        this.updatedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigVersionUpdatedEvent(this, updatedBy));
    }

    /**
     * 校验配置数据
     */
    public boolean validate(ConfigSchema schema) {
        // 根据 schema 校验 configData
        return schema.validate(this.configData);
    }

    private static String generateVersionNumber() {
        return "cfg-" + LocalDate.now().format(DateTimeFormatter.BASIC_ISO_DATE)
            + "-" + String.format("%03d", new Random().nextInt(1000));
    }
}

/**
 * 配置状态
 */
public enum ConfigStatus {
    DRAFT,       // 草稿
    PUBLISHED,   // 已发布
    DEPRECATED   // 已废弃
}
```

### 2.2 值对象

```java
/**
 * 目标范围
 */
public class TargetScope {
    private List<String> storeIds;
    private List<String> profileIds;
    private List<String> terminalIds;
    private List<String> tags;

    public boolean matches(TerminalInstance terminal) {
        if (storeIds != null && !storeIds.isEmpty()) {
            if (!storeIds.contains(terminal.getStoreId())) {
                return false;
            }
        }
        if (profileIds != null && !profileIds.isEmpty()) {
            if (!profileIds.contains(terminal.getProfileId())) {
                return false;
            }
        }
        if (terminalIds != null && !terminalIds.isEmpty()) {
            if (!terminalIds.contains(terminal.getTerminalId())) {
                return false;
            }
        }
        if (tags != null && !tags.isEmpty()) {
            if (terminal.getTags() == null ||
                !terminal.getTags().stream().anyMatch(tags::contains)) {
                return false;
            }
        }
        return true;
    }
}

/**
 * 灰度策略
 */
public class GrayscaleStrategy {
    private GrayscaleType type;
    private Integer percentage;
    private List<String> terminalIds;
    private List<Integer> batches;

    public List<TerminalInstance> selectTargets(List<TerminalInstance> allTerminals) {
        switch (type) {
            case PERCENTAGE:
                return selectByPercentage(allTerminals);
            case TERMINAL_LIST:
                return selectByTerminalList(allTerminals);
            case BATCH:
                return selectByBatch(allTerminals);
            default:
                return allTerminals;
        }
    }

    private List<TerminalInstance> selectByPercentage(List<TerminalInstance> allTerminals) {
        int targetCount = (int) Math.ceil(allTerminals.size() * percentage / 100.0);
        Collections.shuffle(allTerminals);
        return allTerminals.subList(0, Math.min(targetCount, allTerminals.size()));
    }

    private List<TerminalInstance> selectByTerminalList(List<TerminalInstance> allTerminals) {
        return allTerminals.stream()
            .filter(t -> terminalIds.contains(t.getTerminalId()))
            .collect(Collectors.toList());
    }

    private List<TerminalInstance> selectByBatch(List<TerminalInstance> allTerminals) {
        // 按批次分批发布
        // 实现略
        return allTerminals;
    }
}

public enum GrayscaleType {
    ALL,           // 全量发布
    PERCENTAGE,    // 按百分比
    TERMINAL_LIST, // 按终端列表
    BATCH          // 按批次
}
```

## 3. ConfigRelease 聚合根

```java
@Entity
@Table(name = "config_releases")
public class ConfigRelease {

    @Id
    private String releaseId;

    private String configVersionId;
    private String releaseName;

    // 目标终端
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private TargetScope targetScope;

    // 灰度策略
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private GrayscaleStrategy grayscaleStrategy;

    @Enumerated(EnumType.STRING)
    private ReleaseStatus status;  // PENDING, IN_PROGRESS, COMPLETED, FAILED, ROLLED_BACK

    private LocalDateTime scheduledAt;
    private LocalDateTime startedAt;
    private LocalDateTime completedAt;

    private Integer totalCount;
    private Integer successCount;
    private Integer failedCount;
    private Integer pendingCount;

    // 回滚信息
    private String rollbackReason;
    private LocalDateTime rolledBackAt;

    private String createdBy;
    private LocalDateTime createdAt;

    @Version
    private Integer version;

    // 领域方法

    /**
     * 创建配置发布
     */
    public static ConfigRelease create(CreateConfigReleaseCommand command) {
        ConfigRelease release = new ConfigRelease();
        release.releaseId = IdGenerator.generateReleaseId();
        release.configVersionId = command.getConfigVersionId();
        release.releaseName = command.getReleaseName();
        release.targetScope = command.getTargetScope();
        release.grayscaleStrategy = command.getGrayscaleStrategy();
        release.status = ReleaseStatus.PENDING;
        release.scheduledAt = command.getScheduledAt();
        release.createdBy = command.getCreatedBy();
        release.createdAt = LocalDateTime.now();

        release.addDomainEvent(new ConfigReleaseCreatedEvent(release));
        return release;
    }

    /**
     * 开始发布
     */
    public void start() {
        if (this.status != ReleaseStatus.PENDING) {
            throw new IllegalStateException("只能启动待发布状态的发布单");
        }
        this.status = ReleaseStatus.IN_PROGRESS;
        this.startedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigReleaseStartedEvent(this));
    }

    /**
     * 完成发布
     */
    public void complete() {
        if (this.status != ReleaseStatus.IN_PROGRESS) {
            throw new IllegalStateException("只能完成发布中状态的发布单");
        }
        this.status = ReleaseStatus.COMPLETED;
        this.completedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigReleaseCompletedEvent(this));
    }

    /**
     * 发布失败
     */
    public void fail(String reason) {
        this.status = ReleaseStatus.FAILED;
        this.completedAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigReleaseFailedEvent(this, reason));
    }

    /**
     * 回滚发布
     */
    public void rollback(String reason) {
        if (this.status != ReleaseStatus.IN_PROGRESS &&
            this.status != ReleaseStatus.COMPLETED) {
            throw new IllegalStateException("只能回滚发布中或已完成状态的发布单");
        }
        this.status = ReleaseStatus.ROLLED_BACK;
        this.rollbackReason = reason;
        this.rolledBackAt = LocalDateTime.now();
        this.addDomainEvent(new ConfigReleaseRolledBackEvent(this, reason));
    }

    /**
     * 更新统计
     */
    public void updateStats(int total, int success, int failed) {
        this.totalCount = total;
        this.successCount = success;
        this.failedCount = failed;
        this.pendingCount = total - success - failed;

        // 检查是否全部完成
        if (this.pendingCount == 0 && this.status == ReleaseStatus.IN_PROGRESS) {
            if (this.failedCount == 0) {
                this.complete();
            } else if (this.successCount == 0) {
                this.fail("所有终端执行失败");
            } else {
                this.complete();
            }
        }
    }

    /**
     * 计算成功率
     */
    public double getSuccessRate() {
        if (totalCount == null || totalCount == 0) {
            return 0.0;
        }
        return (double) successCount / totalCount * 100;
    }
}

/**
 * 发布状态
 */
public enum ReleaseStatus {
    PENDING,      // 待发布
    IN_PROGRESS,  // 发布中
    COMPLETED,    // 已完成
    FAILED,       // 失败
    ROLLED_BACK   // 已回滚
}
```

## 5. 配置管理应用服务

### 5.1 ConfigApplicationService

```java
@Service
@Transactional
public class ConfigApplicationService {

    @Autowired
    private ConfigVersionRepository configVersionRepository;

    @Autowired
    private ConfigReleaseRepository configReleaseRepository;

    @Autowired
    private ConfigTaskInstanceRepository configTaskInstanceRepository;

    @Autowired
    private TerminalInstanceRepository terminalInstanceRepository;

    @Autowired
    private TdpClient tdpClient;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 创建配置版本
     */
    public CreateConfigVersionResponse createConfigVersion(
            CreateConfigVersionRequest request) {

        // 1. 校验配置数据
        ConfigSchema schema = configSchemaRepository
            .findById(request.getSchemaId())
            .orElseThrow(() -> new ConfigSchemaNotFoundException());

        if (!schema.validate(request.getConfigData())) {
            throw new ConfigValidationException("配置数据校验失败");
        }

        // 2. 创建配置版本
        ConfigVersion configVersion = ConfigVersion.create(
            CreateConfigVersionCommand.builder()
                .configName(request.getConfigName())
                .schemaId(request.getSchemaId())
                .projectId(request.getProjectId())
                .profileId(request.getProfileId())
                .configData(request.getConfigData())
                .changeLog(request.getChangeLog())
                .baselineVersionId(request.getBaselineVersionId())
                .createdBy(request.getCreatedBy())
                .build()
        );

        configVersionRepository.save(configVersion);

        // 3. 发布领域事件
        eventPublisher.publish(configVersion.getDomainEvents());

        return CreateConfigVersionResponse.builder()
            .configVersionId(configVersion.getConfigVersionId())
            .version(configVersion.getVersion())
            .build();
    }

    /**
     * 创建配置发布
     */
    public CreateConfigReleaseResponse createConfigRelease(
            CreateConfigReleaseRequest request) {

        // 1. 查询配置版本
        ConfigVersion configVersion = configVersionRepository
            .findById(request.getConfigVersionId())
            .orElseThrow(() -> new ConfigVersionNotFoundException());

        if (configVersion.getStatus() != ConfigStatus.PUBLISHED) {
            throw new IllegalStateException("只能发布已发布状态的配置版本");
        }

        // 2. 创建发布单
        ConfigRelease release = ConfigRelease.create(
            CreateConfigReleaseCommand.builder()
                .configVersionId(configVersion.getConfigVersionId())
                .releaseName(request.getReleaseName())
                .targetScope(request.getTargetScope())
                .grayscaleStrategy(request.getGrayscaleStrategy())
                .scheduledAt(request.getScheduledAt())
                .createdBy(request.getCreatedBy())
                .build()
        );

        configReleaseRepository.save(release);

        // 3. 查询目标终端
        List<TerminalInstance> allTerminals = terminalInstanceRepository
            .findByTargetScope(request.getTargetScope());

        // 4. 应用灰度策略
        List<TerminalInstance> targetTerminals = request.getGrayscaleStrategy()
            .selectTargets(allTerminals);

        // 5. 创建任务实例
        for (TerminalInstance terminal : targetTerminals) {
            ConfigTaskInstance taskInstance = ConfigTaskInstance.create(
                release.getReleaseId(),
                terminal.getTerminalId(),
                configVersion.getConfigVersionId()
            );
            configTaskInstanceRepository.save(taskInstance);
        }

        release.updateStats(targetTerminals.size(), 0, 0);
        configReleaseRepository.save(release);

        // 6. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());

        return CreateConfigReleaseResponse.builder()
            .releaseId(release.getReleaseId())
            .totalCount(targetTerminals.size())
            .build();
    }

    /**
     * 执行配置发布
     */
    public void executeConfigRelease(String releaseId) {
        // 1. 查询发布单
        ConfigRelease release = configReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new ConfigReleaseNotFoundException());

        // 2. 开始发布
        release.start();
        configReleaseRepository.save(release);

        // 3. 查询配置版本
        ConfigVersion configVersion = configVersionRepository
            .findById(release.getConfigVersionId())
            .orElseThrow(() -> new ConfigVersionNotFoundException());

        // 4. 查询任务实例
        List<ConfigTaskInstance> taskInstances = configTaskInstanceRepository
            .findByReleaseId(releaseId);

        // 5. 委托 TDP 投递配置任务
        for (ConfigTaskInstance taskInstance : taskInstances) {
            tdpClient.publishConfigTask(TdpConfigTask.builder()
                .taskId(taskInstance.getTaskInstanceId())
                .terminalId(taskInstance.getTerminalId())
                .configVersionId(configVersion.getConfigVersionId())
                .configData(configVersion.getConfigData())
                .priority(TaskPriority.NORMAL)
                .expiresAt(LocalDateTime.now().plusDays(1))
                .build());
        }

        // 6. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());
    }

    /**
     * 回报配置应用结果
     */
    public void reportConfigResult(ReportConfigResultRequest request) {
        // 1. 查询任务实例
        ConfigTaskInstance taskInstance = configTaskInstanceRepository
            .findById(request.getTaskInstanceId())
            .orElseThrow(() -> new ConfigTaskInstanceNotFoundException());

        // 2. 更新任务实例
        if (request.getStatus() == TaskInstanceStatus.SUCCESS) {
            taskInstance.markSuccess(request.getResult());
        } else {
            taskInstance.markFailed(request.getErrorMessage());
        }
        configTaskInstanceRepository.save(taskInstance);

        // 3. 更新发布单统计
        ConfigRelease release = configReleaseRepository
            .findById(taskInstance.getReleaseId())
            .orElseThrow(() -> new ConfigReleaseNotFoundException());

        int successCount = configTaskInstanceRepository
            .countByReleaseIdAndStatus(
                taskInstance.getReleaseId(),
                TaskInstanceStatus.SUCCESS
            );
        int failedCount = configTaskInstanceRepository
            .countByReleaseIdAndStatus(
                taskInstance.getReleaseId(),
                TaskInstanceStatus.FAILED
            );

        release.updateStats(release.getTotalCount(), successCount, failedCount);
        configReleaseRepository.save(release);

        // 4. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());
    }

    /**
     * 回滚配置发布
     */
    public void rollbackConfigRelease(String releaseId, String reason) {
        // 1. 查询发布单
        ConfigRelease release = configReleaseRepository
            .findById(releaseId)
            .orElseThrow(() -> new ConfigReleaseNotFoundException());

        // 2. 查询基线配置版本
        ConfigVersion currentVersion = configVersionRepository
            .findById(release.getConfigVersionId())
            .orElseThrow(() -> new ConfigVersionNotFoundException());

        if (currentVersion.getBaselineVersionId() == null) {
            throw new IllegalStateException("没有基线版本，无法回滚");
        }

        ConfigVersion baselineVersion = configVersionRepository
            .findById(currentVersion.getBaselineVersionId())
            .orElseThrow(() -> new ConfigVersionNotFoundException());

        // 3. 创建回滚发布单
        ConfigRelease rollbackRelease = ConfigRelease.create(
            CreateConfigReleaseCommand.builder()
                .configVersionId(baselineVersion.getConfigVersionId())
                .releaseName("回滚-" + release.getReleaseName())
                .targetScope(release.getTargetScope())
                .grayscaleStrategy(new GrayscaleStrategy(GrayscaleType.ALL))
                .scheduledAt(LocalDateTime.now())
                .createdBy("system")
                .build()
        );

        configReleaseRepository.save(rollbackRelease);

        // 4. 标记原发布单为已回滚
        release.rollback(reason);
        configReleaseRepository.save(release);

        // 5. 执行回滚发布
        executeConfigRelease(rollbackRelease.getReleaseId());

        // 6. 发布领域事件
        eventPublisher.publish(release.getDomainEvents());
        eventPublisher.publish(rollbackRelease.getDomainEvents());
    }
}
```

## 7. 配置 Schema 管理

### 7.1 ConfigSchema 实体

```java
@Entity
@Table(name = "config_schemas")
public class ConfigSchema {

    @Id
    private String schemaId;

    private String schemaName;
    private String version;

    // Schema 定义（JSON Schema）
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private Map<String, Object> schemaDefinition;

    /**
     * 校验配置数据
     */
    public boolean validate(Map<String, Object> configData) {
        // 使用 JSON Schema 校验
        JsonSchema jsonSchema = JsonSchemaFactory.getInstance()
            .getSchema(schemaDefinition);

        Set<ValidationMessage> errors = jsonSchema.validate(
            new ObjectMapper().valueToTree(configData)
        );

        return errors.isEmpty();
    }
}
```

## 8. 本文结论

配置管理服务设计展示了完整的配置发布实现：

- **ConfigVersion**：配置版本聚合根，支持创建、发布、废弃、更新
- **ConfigRelease**：配置发布聚合根，支持灰度发布、回滚、统计
- **ConfigTaskInstance**：配置任务实例，追踪每个终端的执行状态
- **ConfigApplicationService**：配置管理应用服务，完整的发布流程
- **灰度策略**：支持全量、按百分比、按终端列表、按批次
- **配置回滚**：支持回滚到基线版本
- **Schema 校验**：基于 JSON Schema 的配置数据校验
- **REST API**：完整的配置管理接口
