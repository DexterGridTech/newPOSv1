# 05-远程控制服务设计

## 1. 远程控制概述

远程控制服务负责对终端设备执行运维操作，包括：
- 远程重启
- 日志上传
- 强制同步
- 屏幕锁定
- 清除缓存
- 诊断信息收集

## 2. ControlTask 聚合根

### 2.1 领域模型

```java
@Entity
@Table(name = "control_tasks")
public class ControlTask {

    @Id
    private String taskId;  // 任务唯一标识

    private String taskName;  // 任务名称，如"重启所有POS终端"

    @Enumerated(EnumType.STRING)
    private ControlCommand command;  // 控制命令类型

    // 目标终端范围
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private TargetScope targetScope;  // 目标终端范围（门店、设备模板等）

    // 命令参数
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private Map<String, Object> parameters;  // 命令执行参数，不同命令有不同参数

    @Enumerated(EnumType.STRING)
    private TaskStatus status;  // 任务状态

    // 优先级
    @Enumerated(EnumType.STRING)
    private TaskPriority priority;  // 任务优先级：URGENT（紧急）, HIGH（高）, NORMAL（普通）, LOW（低）

    // 时间信息
    private LocalDateTime scheduledAt;  // 计划执行时间
    private LocalDateTime startedAt;  // 实际开始时间
    private LocalDateTime completedAt;  // 完成时间
    private LocalDateTime expiresAt;  // 过期时间，超过此时间任务自动失效

    // 统计信息
    private Integer totalCount;  // 总终端数
    private Integer successCount;  // 成功数
    private Integer failedCount;  // 失败数
    private Integer pendingCount;  // 待执行数

    // 失败信息
    private String failureReason;  // 任务失败原因

    // 审计信息
    private String createdBy;  // 创建人
    private LocalDateTime createdAt;  // 创建时间

    // 版本号（乐观锁）
    @Version
    private Integer version;

    /**
     * 创建控制任务
     */
    public static ControlTask create(CreateControlTaskCommand command) {
        // 1. 参数校验
        if (command.getTaskName() == null || command.getTaskName().isEmpty()) {
            throw new IllegalArgumentException("任务名称不能为空");
        }
        if (command.getCommand() == null) {
            throw new IllegalArgumentException("控制命令不能为空");
        }
        if (command.getTargetScope() == null) {
            throw new IllegalArgumentException("目标范围不能为空");
        }

        // 2. 校验命令参数
        validateCommandParameters(command.getCommand(), command.getParameters());

        // 3. 创建任务实例
        ControlTask task = new ControlTask();
        task.taskId = IdGenerator.generateTaskId();
        task.taskName = command.getTaskName();
        task.command = command.getCommand();
        task.targetScope = command.getTargetScope();
        task.parameters = command.getParameters() != null ?
            command.getParameters() : new HashMap<>();
        task.status = TaskStatus.PENDING;
        task.priority = command.getPriority() != null ?
            command.getPriority() : TaskPriority.NORMAL;
        task.scheduledAt = command.getScheduledAt() != null ?
            command.getScheduledAt() : LocalDateTime.now();
        task.expiresAt = command.getExpiresAt() != null ?
            command.getExpiresAt() : LocalDateTime.now().plusDays(1);
        task.createdBy = command.getCreatedBy();
        task.createdAt = LocalDateTime.now();

        // 4. 发布领域事件
        task.addDomainEvent(new ControlTaskCreatedEvent(task));
        return task;
    }

    /**
     * 开始执行
     */
    public void start() {
        if (this.status != TaskStatus.PENDING) {
            throw new IllegalStateException(
                "只能启动待执行状态的任务，当前状态：" + this.status
            );
        }

        // 检查是否过期
        if (LocalDateTime.now().isAfter(this.expiresAt)) {
            throw new IllegalStateException("任务已过期，无法执行");
        }

        this.status = TaskStatus.IN_PROGRESS;
        this.startedAt = LocalDateTime.now();
        this.addDomainEvent(new ControlTaskStartedEvent(this));
    }

    /**
     * 完成任务
     */
    public void complete() {
        if (this.status != TaskStatus.IN_PROGRESS) {
            throw new IllegalStateException(
                "只能完成执行中状态的任务，当前状态：" + this.status
            );
        }

        this.status = TaskStatus.COMPLETED;
        this.completedAt = LocalDateTime.now();
        this.addDomainEvent(new ControlTaskCompletedEvent(this));
    }

    /**
     * 任务失败
     */
    public void fail(String reason) {
        if (this.status == TaskStatus.COMPLETED) {
            throw new IllegalStateException("已完成的任务不能标记为失败");
        }

        this.status = TaskStatus.FAILED;
        this.completedAt = LocalDateTime.now();
        this.failureReason = reason;
        this.addDomainEvent(new ControlTaskFailedEvent(this, reason));
    }

    /**
     * 更新统计
     */
    public void updateStats(int total, int success, int failed) {
        this.totalCount = total;
        this.successCount = success;
        this.failedCount = failed;
        this.pendingCount = total - success - failed;

        // 检查是否全部完成
        if (this.pendingCount == 0 && this.status == TaskStatus.IN_PROGRESS) {
            if (this.failedCount == 0) {
                this.complete();
            } else if (this.successCount == 0) {
                this.fail("所有终端执行失败");
            } else {
                this.complete();
            }
        }
    }

    /**
     * 计算成功率
     */
    public double getSuccessRate() {
        if (totalCount == null || totalCount == 0) {
            return 0.0;
        }
        return (double) successCount / totalCount * 100;
    }

    /**
     * 是否可以取消
     */
    public boolean canCancel() {
        return this.status == TaskStatus.PENDING ||
               this.status == TaskStatus.IN_PROGRESS;
    }

    /**
     * 取消任务
     */
    public void cancel(String reason) {
        if (!canCancel()) {
            throw new IllegalStateException(
                "当前状态不允许取消：" + this.status
            );
        }

        this.status = TaskStatus.CANCELLED;
        this.completedAt = LocalDateTime.now();
        this.failureReason = reason;
        this.addDomainEvent(new ControlTaskCancelledEvent(this, reason));
    }

    /**
     * 校验命令参数
     */
    private static void validateCommandParameters(
            ControlCommand command,
            Map<String, Object> parameters) {

        if (parameters == null) {
            parameters = new HashMap<>();
        }

        switch (command) {
            case RESTART:
                // 重启命令可选参数：delay（延迟时间，毫秒）
                if (parameters.containsKey("delay")) {
                    Integer delay = (Integer) parameters.get("delay");
                    if (delay < 0 || delay > 300000) {
                        throw new IllegalArgumentException(
                            "重启延迟时间必须在 0-300000 毫秒之间"
                        );
                    }
                }
                break;

            case LOG_UPLOAD:
                // 日志上传命令必需参数：logLevel, lastHours
                if (!parameters.containsKey("logLevel")) {
                    throw new IllegalArgumentException("日志上传命令缺少 logLevel 参数");
                }
                if (!parameters.containsKey("lastHours")) {
                    throw new IllegalArgumentException("日志上传命令缺少 lastHours 参数");
                }
                Integer lastHours = (Integer) parameters.get("lastHours");
                if (lastHours < 1 || lastHours > 72) {
                    throw new IllegalArgumentException(
                        "日志时间范围必须在 1-72 小时之间"
                    );
                }
                break;

            case SCREEN_LOCK:
                // 屏幕锁定命令可选参数：duration（锁定时长，秒）
                if (parameters.containsKey("duration")) {
                    Integer duration = (Integer) parameters.get("duration");
                    if (duration < 60 || duration > 3600) {
                        throw new IllegalArgumentException(
                            "锁定时长必须在 60-3600 秒之间"
                        );
                    }
                }
                break;

            case CLEAR_CACHE:
                // 清除缓存命令可选参数：cacheType（缓存类型）
                if (parameters.containsKey("cacheType")) {
                    String cacheType = (String) parameters.get("cacheType");
                    if (!Arrays.asList("ALL", "IMAGE", "DATA", "TEMP").contains(cacheType)) {
                        throw new IllegalArgumentException(
                            "缓存类型必须是 ALL, IMAGE, DATA, TEMP 之一"
                        );
                    }
                }
                break;

            case DIAGNOSTICS:
                // 诊断命令可选参数：includeScreenshot（是否包含截图）
                break;

            case FORCE_SYNC:
                // 强制同步命令无需参数
                break;

            default:
                throw new IllegalArgumentException("不支持的控制命令：" + command);
        }
    }
}
```

## 3. TaskExecutionRecord 实体

```java
@Entity
@Table(name = "task_execution_records")
public class TaskExecutionRecord {

    @Id
    private String recordId;  // 执行记录唯一标识

    private String taskId;  // 关联的任务ID
    private String terminalId;  // 目标终端ID

    @Enumerated(EnumType.STRING)
    private ExecutionStatus status;  // 执行状态

    // 时间信息
    private LocalDateTime createdAt;  // 创建时间
    private LocalDateTime executedAt;  // 执行时间

    // 执行结果
    @Type(JsonBinaryType.class)
    @Column(columnDefinition = "jsonb")
    private Map<String, Object> result;  // 执行结果详情

    // 错误信息
    private String errorCode;  // 错误码
    private String errorMessage;  // 错误信息

    // 重试信息
    private Integer retryCount;  // 重试次数
    private LocalDateTime lastRetryAt;  // 最后重试时间

    /**
     * 创建执行记录
     */
    public static TaskExecutionRecord create(
            String taskId, String terminalId) {

        TaskExecutionRecord record = new TaskExecutionRecord();
        record.recordId = IdGenerator.generateRecordId();
        record.taskId = taskId;
        record.terminalId = terminalId;
        record.status = ExecutionStatus.PENDING;
        record.createdAt = LocalDateTime.now();
        record.retryCount = 0;
        return record;
    }

    /**
     * 标记成功
     */
    public void markSuccess(Map<String, Object> result) {
        if (this.status == ExecutionStatus.SUCCESS) {
            throw new IllegalStateException("执行记录已标记为成功");
        }

        this.status = ExecutionStatus.SUCCESS;
        this.executedAt = LocalDateTime.now();
        this.result = result;
        this.errorCode = null;
        this.errorMessage = null;
    }

    /**
     * 标记失败
     */
    public void markFailed(String errorCode, String errorMessage) {
        if (this.status == ExecutionStatus.SUCCESS) {
            throw new IllegalStateException("成功的执行记录不能标记为失败");
        }

        this.status = ExecutionStatus.FAILED;
        this.executedAt = LocalDateTime.now();
        this.errorCode = errorCode;
        this.errorMessage = errorMessage;
    }

    /**
     * 标记超时
     */
    public void markTimeout() {
        if (this.status == ExecutionStatus.SUCCESS) {
            throw new IllegalStateException("成功的执行记录不能标记为超时");
        }

        this.status = ExecutionStatus.TIMEOUT;
        this.executedAt = LocalDateTime.now();
        this.errorCode = "EXECUTION_TIMEOUT";
        this.errorMessage = "任务执行超时";
    }

    /**
     * 重试
     */
    public void retry() {
        if (this.status == ExecutionStatus.SUCCESS) {
            throw new IllegalStateException("成功的执行记录不能重试");
        }

        this.status = ExecutionStatus.PENDING;
        this.retryCount++;
        this.lastRetryAt = LocalDateTime.now();
        this.errorCode = null;
        this.errorMessage = null;
    }

    /**
     * 是否可以重试
     */
    public boolean canRetry(int maxRetryCount) {
        return this.status == ExecutionStatus.FAILED &&
               this.retryCount < maxRetryCount;
    }
}

/**
 * 执行状态
 */
public enum ExecutionStatus {
    PENDING("待执行", "等待终端执行"),
    SUCCESS("成功", "执行成功"),
    FAILED("失败", "执行失败"),
    TIMEOUT("超时", "执行超时");

    private final String displayName;
    private final String description;

    ExecutionStatus(String displayName, String description) {
        this.displayName = displayName;
        this.description = description;
    }

    public String getDisplayName() {
        return displayName;
    }

    public String getDescription() {
        return description;
    }
}
```

## 4. 远程控制应用服务

```java
@Service
@Transactional
public class ControlTaskApplicationService {

    @Autowired
    private ControlTaskRepository controlTaskRepository;

    @Autowired
    private TaskExecutionRecordRepository taskExecutionRecordRepository;

    @Autowired
    private TerminalInstanceRepository terminalInstanceRepository;

    @Autowired
    private DeviceProfileRepository deviceProfileRepository;

    @Autowired
    private ControlTaskAuditLogRepository auditLogRepository;

    @Autowired
    private TdpClient tdpClient;

    @Autowired
    private DomainEventPublisher eventPublisher;

    /**
     * 创建控制任务
     */
    public CreateControlTaskResponse createControlTask(
            CreateControlTaskRequest request) {

        // 1. 校验命令权限
        validateCommandPermission(request.getCommand(), request.getTargetScope());

        // 2. 创建控制任务
        ControlTask task = ControlTask.create(CreateControlTaskCommand.builder()
            .taskName(request.getTaskName())
            .command(request.getCommand())
            .targetScope(request.getTargetScope())
            .parameters(request.getParameters())
            .priority(request.getPriority())
            .scheduledAt(request.getScheduledAt())
            .expiresAt(request.getExpiresAt())
            .createdBy(request.getCreatedBy())
            .build());

        controlTaskRepository.save(task);

        // 3. 查询目标终端
        List<TerminalInstance> terminals = terminalInstanceRepository
            .findByTargetScope(request.getTargetScope());

        if (terminals.isEmpty()) {
            throw new NoTargetTerminalsException("没有找到符合条件的目标终端");
        }

        // 4. 过滤支持该命令的终端
        List<TerminalInstance> supportedTerminals = filterSupportedTerminals(
            terminals,
            request.getCommand()
        );

        if (supportedTerminals.isEmpty()) {
            throw new NoSupportedTerminalsException(
                "没有终端支持该命令：" + request.getCommand()
            );
        }

        // 5. 创建执行记录
        for (TerminalInstance terminal : supportedTerminals) {
            TaskExecutionRecord record = TaskExecutionRecord.create(
                task.getTaskId(),
                terminal.getTerminalId()
            );
            taskExecutionRecordRepository.save(record);
        }

        // 6. 更新任务统计
        task.updateStats(supportedTerminals.size(), 0, 0);
        controlTaskRepository.save(task);

        // 7. 发布领域事件
        eventPublisher.publish(task.getDomainEvents());

        // 8. 记录审计日志
        log.info("创建控制任务：taskId={}, command={}, targetCount={}",
            task.getTaskId(),
            task.getCommand(),
            supportedTerminals.size());

        return CreateControlTaskResponse.builder()
            .taskId(task.getTaskId())
            .taskName(task.getTaskName())
            .command(task.getCommand())
            .status(task.getStatus())
            .totalCount(supportedTerminals.size())
            .createdAt(task.getCreatedAt())
            .build();
    }

    /**
     * 执行控制任务
     */
    public void executeControlTask(String taskId) {
        // 1. 查询任务
        ControlTask task = controlTaskRepository
            .findById(taskId)
            .orElseThrow(() -> new ControlTaskNotFoundException());

        // 2. 开始执行
        task.start();
        controlTaskRepository.save(task);

        // 3. 查询执行记录
        List<TaskExecutionRecord> records = taskExecutionRecordRepository
            .findByTaskIdAndStatus(taskId, ExecutionStatus.PENDING);

        // 4. 委托 TDP 投递控制命令
        int successCount = 0;
        for (TaskExecutionRecord record : records) {
            try {
                tdpClient.publishControlCommand(TdpControlCommand.builder()
                    .taskId(task.getTaskId())
                    .recordId(record.getRecordId())
                    .terminalId(record.getTerminalId())
                    .command(task.getCommand())
                    .parameters(task.getParameters())
                    .priority(task.getPriority())
                    .expiresAt(task.getExpiresAt())
                    .build());

                successCount++;
            } catch (Exception e) {
                log.error("投递控制命令失败：recordId={}, terminalId={}",
                    record.getRecordId(),
                    record.getTerminalId(),
                    e);

                // 标记执行记录为失败
                record.markFailed("TDP_PUBLISH_FAILED", "投递命令失败：" + e.getMessage());
                taskExecutionRecordRepository.save(record);
            }
        }

        // 5. 发布领域事件
        eventPublisher.publish(task.getDomainEvents());

        log.info("执行控制任务：taskId={}, totalRecords={}, successCount={}",
            taskId,
            records.size(),
            successCount);
    }

    /**
     * 回报任务执行结果
     */
    public void reportTaskResult(ReportTaskResultRequest request) {
        // 1. 查询执行记录
        TaskExecutionRecord record = taskExecutionRecordRepository
            .findByTaskIdAndTerminalId(
                request.getTaskId(),
                request.getTerminalId()
            )
            .orElseThrow(() -> new TaskExecutionRecordNotFoundException());

        // 2. 更新执行记录
        if (request.getStatus() == ExecutionStatus.SUCCESS) {
            record.markSuccess(request.getResult());
        } else if (request.getStatus() == ExecutionStatus.TIMEOUT) {
            record.markTimeout();
        } else {
            record.markFailed(request.getErrorCode(), request.getErrorMessage());
        }
        taskExecutionRecordRepository.save(record);

        // 3. 查询任务
        ControlTask task = controlTaskRepository
            .findById(request.getTaskId())
            .orElseThrow(() -> new ControlTaskNotFoundException());

        // 4. 更新任务统计
        int successCount = taskExecutionRecordRepository
            .countByTaskIdAndStatus(request.getTaskId(), ExecutionStatus.SUCCESS);
        int failedCount = taskExecutionRecordRepository
            .countByTaskIdAndStatus(request.getTaskId(), ExecutionStatus.FAILED);
        int timeoutCount = taskExecutionRecordRepository
            .countByTaskIdAndStatus(request.getTaskId(), ExecutionStatus.TIMEOUT);

        task.updateStats(task.getTotalCount(), successCount, failedCount + timeoutCount);
        controlTaskRepository.save(task);

        // 5. 创建审计日志
        ControlTaskAuditLog auditLog = ControlTaskAuditLog.create(
            request.getTaskId(),
            request.getTerminalId(),
            task.getCommand(),
            task.getCreatedBy(),
            task.getParameters(),
            request.getResult()
        );
        auditLogRepository.save(auditLog);

        // 6. 发布领域事件
        eventPublisher.publish(task.getDomainEvents());

        log.info("回报任务执行结果：taskId={}, terminalId={}, status={}",
            request.getTaskId(),
            request.getTerminalId(),
            request.getStatus());
    }

    /**
     * 取消控制任务
     */
    public void cancelControlTask(String taskId, String reason) {
        // 1. 查询任务
        ControlTask task = controlTaskRepository
            .findById(taskId)
            .orElseThrow(() -> new ControlTaskNotFoundException());

        // 2. 取消任务
        task.cancel(reason);
        controlTaskRepository.save(task);

        // 3. 更新待执行的记录为失败
        List<TaskExecutionRecord> pendingRecords = taskExecutionRecordRepository
            .findByTaskIdAndStatus(taskId, ExecutionStatus.PENDING);

        for (TaskExecutionRecord record : pendingRecords) {
            record.markFailed("TASK_CANCELLED", "任务已取消：" + reason);
            taskExecutionRecordRepository.save(record);
        }

        // 4. 发布领域事件
        eventPublisher.publish(task.getDomainEvents());

        log.info("取消控制任务：taskId={}, reason={}", taskId, reason);
    }

    /**
     * 重试失败的执行记录
     */
    public RetryTaskResponse retryFailedRecords(String taskId) {
        // 1. 查询任务
        ControlTask task = controlTaskRepository
            .findById(taskId)
            .orElseThrow(() -> new ControlTaskNotFoundException());

        // 2. 查询失败的执行记录
        List<TaskExecutionRecord> failedRecords = taskExecutionRecordRepository
            .findByTaskIdAndStatus(taskId, ExecutionStatus.FAILED);

        if (failedRecords.isEmpty()) {
            throw new NoFailedRecordsException("没有失败的执行记录");
        }

        // 3. 过滤可重试的记录（最多重试3次）
        List<TaskExecutionRecord> retryableRecords = failedRecords.stream()
            .filter(r -> r.canRetry(3))
            .collect(Collectors.toList());

        if (retryableRecords.isEmpty()) {
            throw new NoRetryableRecordsException("没有可重试的执行记录");
        }

        // 4. 重试
        int retryCount = 0;
        for (TaskExecutionRecord record : retryableRecords) {
            try {
                // 标记为待执行
                record.retry();
                taskExecutionRecordRepository.save(record);

                // 重新投递命令
                tdpClient.publishControlCommand(TdpControlCommand.builder()
                    .taskId(task.getTaskId())
                    .recordId(record.getRecordId())
                    .terminalId(record.getTerminalId())
                    .command(task.getCommand())
                    .parameters(task.getParameters())
                    .priority(TaskPriority.HIGH)  // 重试任务提高优先级
                    .expiresAt(task.getExpiresAt())
                    .build());

                retryCount++;
            } catch (Exception e) {
                log.error("重试失败：recordId={}", record.getRecordId(), e);
            }
        }

        log.info("重试失败记录：taskId={}, retryCount={}", taskId, retryCount);

        return RetryTaskResponse.builder()
            .taskId(taskId)
            .retryCount(retryCount)
            .build();
    }

    /**
     * 校验命令权限
     */
    private void validateCommandPermission(
            ControlCommand command,
            TargetScope targetScope) {

        // 查询目标终端的设备能力模板
        List<TerminalInstance> terminals = terminalInstanceRepository
            .findByTargetScope(targetScope);

        if (terminals.isEmpty()) {
            return;
        }

        // 检查设备模板是否允许该命令
        for (TerminalInstance terminal : terminals) {
            DeviceProfile profile = deviceProfileRepository
                .findById(terminal.getProfileId())
                .orElse(null);

            if (profile != null) {
                if (!profile.getAllowedRemoteCommands().contains(command)) {
                    throw new CommandNotAllowedException(
                        "设备模板不允许该命令：" + command +
                        ", profileId=" + terminal.getProfileId()
                    );
                }
            }
        }
    }

    /**
     * 过滤支持该命令的终端
     */
    private List<TerminalInstance> filterSupportedTerminals(
            List<TerminalInstance> terminals,
            ControlCommand command) {

        return terminals.stream()
            .filter(terminal -> {
                DeviceProfile profile = deviceProfileRepository
                    .findById(terminal.getProfileId())
                    .orElse(null);

                if (profile == null) {
                    return false;
                }

                return profile.getAllowedRemoteCommands().contains(command);
            })
            .collect(Collectors.toList());
    }
}

    private Integer retryCount;
```

```java
/**
 * 重启命令实现
 * 支持延迟重启，避免在关键操作时立即重启
 */
public class RestartCommand implements RemoteCommand {

    private static final Logger log = LoggerFactory.getLogger(RestartCommand.class);

    @Override
    public ControlCommand getCommandType() {
        return ControlCommand.RESTART;
    }

    @Override
    public void validateParameters(Map<String, Object> parameters) {
        if (parameters.containsKey("delay")) {
            Integer delay = (Integer) parameters.get("delay");
            if (delay < 0 || delay > 300000) {
                throw new IllegalArgumentException(
                    "重启延迟时间必须在 0-300000 毫秒之间"
                );
            }
        }
    }

    @Override
    public void execute(TerminalContext context, Map<String, Object> parameters)
            throws CommandExecutionException {
        try {
            // 1. 获取延迟时间（默认 0）
            Integer delay = (Integer) parameters.getOrDefault("delay", 0);

            log.info("执行重启命令：terminalId={}, delay={}ms",
                context.getTerminalId(), delay);

            // 2. 如果有延迟，先等待
            if (delay > 0) {
                log.info("延迟 {} 毫秒后重启", delay);
                Thread.sleep(delay);
            }

            // 3. 记录重启前的状态
            LocalDateTime restartedAt = LocalDateTime.now();

            // 4. 执行重启
            context.restart();

            // 5. 回报结果（注意：重启后此代码可能不会执行）
            Map<String, Object> result = new HashMap<>();
            result.put("restartedAt", restartedAt);
            result.put("delay", delay);
            context.reportResult(result);

        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new CommandExecutionException("重启命令被中断", e);
        } catch (Exception e) {
            log.error("重启命令执行失败：terminalId={}",
                context.getTerminalId(), e);
            context.reportFailure("RESTART_FAILED", e.getMessage());
            throw new CommandExecutionException("重启命令执行失败", e);
        }
    }
}
```

```java
    public ControlCommand getCommandType() {

            throw new IllegalArgumentException("日志上传命令缺少 lastHours 参数");

        if (lastHours < 1 || lastHours > 72) {
            throw new IllegalArgumentException(
                "日志时间范围必须在 1-72 小时之间"
            );
        }

        String logLevel = (String) parameters.get("logLevel");
        if (!Arrays.asList("DEBUG", "INFO", "WARN", "ERROR").contains(logLevel)) {
            throw new IllegalArgumentException(
                "日志级别必须是 DEBUG, INFO, WARN, ERROR 之一"
            );
        }
    }

    @Override
    public void execute(TerminalContext context, Map<String, Object> parameters)
            throws CommandExecutionException {

            // 1. 获取参数
            String logLevel = (String) parameters.get("logLevel");
            Integer lastHours = (Integer) parameters.get("lastHours");

            log.info("执行日志上传命令：terminalId={}, logLevel={}, lastHours={}",
                context.getTerminalId(), logLevel, lastHours);

            // 2. 收集日志
            log.info("开始收集日志...");
            List<LogEntry> logs = context.collectLogs(logLevel, lastHours);
            log.info("收集到 {} 条日志", logs.size());

            // 3. 上传到对象存储
            log.info("开始上传日志...");
            String uploadUrl = context.uploadLogs(logs);
            log.info("日志上传成功：uploadUrl={}", uploadUrl);

            // 4. 回报结果
            Map<String, Object> result = new HashMap<>();
            result.put("uploadUrl", uploadUrl);
            result.put("logCount", logs.size());
            result.put("logLevel", logLevel);
            result.put("lastHours", lastHours);
            result.put("uploadedAt", LocalDateTime.now());
            context.reportResult(result);

        } catch (Exception e) {
            log.error("日志上传命令执行失败：terminalId={}",
                context.getTerminalId(), e);
            context.reportFailure("LOG_UPLOAD_FAILED", e.getMessage());
            throw new CommandExecutionException("日志上传命令执行失败", e);
```

```java
 * 远程控制命令工厂
 * 根据命令类型创建对应的命令实例
 */
@Component
public class RemoteCommandFactory {

    private final Map<ControlCommand, RemoteCommand> commandMap;

    public RemoteCommandFactory() {
        this.commandMap = new HashMap<>();
        this.commandMap.put(ControlCommand.RESTART, new RestartCommand());
        this.commandMap.put(ControlCommand.LOG_UPLOAD, new LogUploadCommand());
        this.commandMap.put(ControlCommand.FORCE_SYNC, new ForceSyncCommand());
        this.commandMap.put(ControlCommand.SCREEN_LOCK, new ScreenLockCommand());
        this.commandMap.put(ControlCommand.CLEAR_CACHE, new ClearCacheCommand());
        this.commandMap.put(ControlCommand.DIAGNOSTICS, new DiagnosticsCommand());
    }

    /**
     * 获取命令实例
     */
    public RemoteCommand getCommand(ControlCommand commandType) {
        RemoteCommand command = commandMap.get(commandType);
        if (command == null) {
            throw new IllegalArgumentException(
                "不支持的控制命令：" + commandType
            );
        }
        return command;
    }

    /**
     * 执行命令
     */
    public void executeCommand(
            ControlCommand commandType,
            TerminalContext context,
            Map<String, Object> parameters) throws CommandExecutionException {

        RemoteCommand command = getCommand(commandType);

        // 1. 校验参数
        command.validateParameters(parameters);

        // 2. 执行命令
        command.execute(context, parameters);
    }
}

/**
 * 命令执行异常
 */
public class CommandExecutionException extends Exception {
    public CommandExecutionException(String message) {
        super(message);
    }

    public CommandExecutionException(String message, Throwable cause) {
        super(message, cause);
    }
}
```
